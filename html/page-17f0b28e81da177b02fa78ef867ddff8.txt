		function validationSuccess(data){
			
			$('#evaluateForBTNNumber').val("");
			if(data['errors'] != undefined || data['systemError'] != undefined)
			{
				/* console.log(data['failureCount']); */
				var failureCount = data['failureCount'];
				/* console.log(data['geoLocation']); */
				$('#addressReturnBtn').removeClass();
				$('#multiAddressErrorBtn').removeClass()
				var geoLocation = data['geoLocation'];
				if(failureCount > 3){
					$('#loading1').hide();
					$('#serviceAddress').hide();
					$('#addressOptions').hide();
					$('#errorContainer').hide();
					$('#showphoneNumber').hide();
					$('#addressValidationContent').show();
					$('#multiAttemptAddressErrorBtn').removeClass();
					$('#multiAttemptAddressErrorBtn').addClass(geoLocation);
					$('#attemptFailure').show().find("div[id$=states]").hide().parent().find('div#'+geoLocation).show();
					if(geoLocation == "13states"){
						WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
		        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + "  Overlay Address Modal Contact Options 13 States");
					}else if(geoLocation == "28states"){
						WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
		        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + "  Overlay Address Modal Contact Options 28 States");
					}else if(geoLocation == "9states"){
						WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
		        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + "  Overlay Address Modal Contact Options 9 States");
					}
					return;
				}else{
					$('#addressErrorContainer').html("");
					
					for( i=0;i<data['errors'].length;i++ ){
						var errorMsg= data['errors'][i].split(':');
						if(errorMsg[1])
							$('#addressErrorContainer').append('<div>'+errorMsg[1]+'</div>');
					}
					$('#addressValidationContent').show();
					$('#loading1').hide();
					
					$('#errorAddress1').text($.trim($('#address1').val()));
					$('#errorAddress2').text($.trim($('#address2').val()));
					$('#errorCitySateZip').text($.trim($('#city').val()) + ", " + $('#state').val() + " " + $('#postalCode').val());
					$('#attemptFailure').hide();
					$('#serviceAddress').hide();
					$('#addressOptions').hide();
					$('#errorContainer').show();
					$('#showphoneNumber').hide();
					
					WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
	        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + " Overlay Address Modal Address Not Found Pg");
							
				}
			}else if(data['showPhoneNumber']=== 'true'){
				$('#loading1').hide();
				$('#attemptFailure').hide();
				$('#serviceAddress').hide();
				$('#addressOptions').hide();
				$('#errorContainer').hide();
				$('#addressValidationContent').show();
				$('#showphoneNumber').show();
				
				WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + " Overlay Address Modal Business Telephone Number Entry");
				
				$('#telephoneNumber_text').focus();
			}else if(!$.isArray(data)){
				var isFamilyPage = $('#FAMILY_PAGE').val();
				var isChildURL = false;
				var targetURL = $('#SeviceEligibiltyLightBox').data('srcURL');
				if(targetURL == undefined)
					targetURL=window.location.href;
				
				var geoLocation = data['geoLocation'];
				
				if(geoLocation == "13states"){
					WebMetrics.Overlay.DCSext.wtB2BStateGroupName = "13 State";
				}else if(geoLocation == "28states"){
					WebMetrics.Overlay.DCSext.wtB2BStateGroupName = "28 State";
				}else if(geoLocation == "9states"){
					WebMetrics.Overlay.DCSext.wtB2BStateGroupName = "9 State";
				}
				WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
				WebMetrics.overlayEventDispatch(WebMetrics.getPageName() +"_Overlay_AddressModalValidation_Complete");
				
				$('.locationOverlay').data('myplugin').closeOverlay();
				$('#processProducts').click();
				var src = "/smallbusiness/common/processing.jsp?successURL="+targetURL;
				$('#processingPage').attr("src",src);
	
			}else if ($.isArray(data)){
				$('#addressOptionsContainer').html("");
				if(data.length >15){
					$('#addressOptionsContainerMain').css('overflow-y','scroll');
					$('#addressOptionsContainerMain').css('max-height','400px');
					}
				for(var count = 0 ; count < data.length ; count++){
					var streetAddressString = data[count]['addressLine1'];
					var city = data[count]['city'];
					var state = data[count]['state'];
					var postalCode = data[count]['postalCode'];
					var addressLine2= data[count]['addressLine2'];
					addressLine2 = $.trim(addressLine2);
					var addressString = "";
					if( addressLine2 == ''){
						addressString = streetAddressString + ", " + city + ", " + state + " " + postalCode;
					} else {
						addressString = streetAddressString + ", " + addressLine2 + ", " + city + ", " + state + " " + postalCode;
					}
					var radioElement = $('<input>').attr({type: 'radio', name: 'addressInput', value: '', id: 'optionalAddressRadio'+count});
						radioElement.data('city',city);
						radioElement.data('state',state);
						radioElement.data('postalCode',postalCode);
						radioElement.data('streetAddress',streetAddressString);
					 	radioElement.data('addressLine2',addressLine2);
						radioElement.data('isRangedAddress','false');
					var radioLabel = $('<span>').html(addressString).addClass('label');
					if(data[count]['isRangedAddress'] == true){
						var streetInput = $('<span style="margin-left:10px;">').append($('<input>').attr({type: 'text', style:'width:80px;',name: 'streetInput', value: '',id: 'streetOption'+count}));
						var streetLabel = $('<span>').html('Updated Street Address').addClass('label');
						var streetElement =$('<div style="display:none;padding:10px;">').attr({id: 'streetElement'+count}).append(streetLabel).append(streetInput);
						radioElement.data('streetElement',streetElement);
						radioElement.data('isRangedAddress',data[count]['isRangedAddress']);
					}
					radioElement.click(function(){   
						if($(this).data('isRangedAddress') == true){
							$('div[id^=streetElement]').each(function(){
								$(this).hide();
							});
							$(this).data('streetElement').show();
						}
						$('input[id^="optionalAddressRadio"]').attr('checked',false);
						$(this).attr('checked',true);
					});
					
					$('#addressValidationContent').show();
					$('#loading1').hide();

					$('#addressOptionsContainer').append($('<div>').append(radioElement).append(radioLabel));
					if(data[count]['isRangedAddress'] == true){
						$('#addressOptionsContainer').append(streetElement);
					}	
					if(count == 0){
						if(data[count]['isRangedAddress'] == true){
							$('#addressReturnBtn').addClass('rangedAddressReturn');
							$('#multiAddressErrorBtn').addClass('rangedAddressReturn');
						}else{
							$('#addressReturnBtn').addClass('multipleAddressReturn');
							$('#multiAddressErrorBtn').addClass('multipleAddressReturn');
						}
						radioElement.attr('checked',true);
						radioElement.trigger('click');	
					}
					
				}
				$('#serviceAddress').hide();
				$('#errorContainer').hide();
				$('#attemptFailure').hide();
				$('#showphoneNumber').hide();
				if((data.length == 1) && (data[0]['isRangedAddress'] != true)){	
					$('#addressOptionsContainerMain').hide();
					var streetAddressString = data[0]['addressLine1'];
					var city = data[0]['city'];
					var state = data[0]['state'];
					var postalCode = data[0]['postalCode'];  
					var addressLine2= data[0]['addressLine2'];
					addressLine2 = $.trim(addressLine2);
					var addressString = "";
					addressString = city + ", " + state + " " + postalCode;
					$('#singleCorrectedAddressDetails').html("");
					$('#singleCorrectedAddressDetails').append($('<div>').append("<strong>"+streetAddressString+"</strong>"));
					if( addressLine2 != ''){
						$('#singleCorrectedAddressDetails').append($('<div>').append("<strong>"+addressLine2+"</strong>"));
					}
					$('#singleCorrectedAddressDetails').append($('<div>').append("<strong>"+addressString+"</strong>"));
					$('#addressReturnBtn').removeClass();
					$('#multiAddressErrorBtn').removeClass();
					$('#addressReturnBtn').addClass('singleReturnAddress');
					$('#multiAddressErrorBtn').addClass('singleReturnAddress')
					$('#singleCorrectedAddress').show();
					
					WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
	        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + " Overlay Address Modal Single Corrected Address Returned");
					
				}else{
					$('#singleCorrectedAddress').hide();
					$('#addressOptionsContainerMain').show();
					WebMetrics.Overlay.DCS.dcsuri = "/smallbusiness/overlay/addressEligibility.jsp";
	        		WebMetrics.overlayPageDispatch(WebMetrics.getPageName() + "  Overlay Address Modal Address Multiple Addresses Returned");
				}
				$('#addressOptions').show();
				
				

			}
