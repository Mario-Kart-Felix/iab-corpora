/**
 * Opens a URL centered in a popup window
 * @param  {string} url    The URL to open
 * @param  {int}    height height of the popup
 * @param  {int}    width  width of the popup
 * @return {bool}        prevent default action
 */
function popitup(url,height,width) {
    var left = (screen.width/2)-(width/2);
    var top = (screen.height/2)-(height/2);
    newwindow=window.open(url,'name','height='+height+',width='+width+',left='+left+',top='+top);
    if (window.focus) {newwindow.focus()}
    return false;
}
