					var monthly_loan_payments = function(principal, months, rate) {
						principal *= 1;
						months    *= 1;
						rate       = 1 * rate || .0001;
						rate      /= 1200;
						
						return ( principal * rate / (1 - (Math.pow( 1 / (1 + rate), months))) ).toFixed(2);
					};
