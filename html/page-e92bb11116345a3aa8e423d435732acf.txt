        /**
         * Restore Default Text from an input field if it does not have a value
         **/
        IC.restore_default_text = function( el ) {
            if( el ) {
                if( IC.is_empty( el ) ) {
                    el.val( el[0].title );
                }
                el.val( IC.trim( el ) );
                el.removeClass('default-text-active').addClass('default-text');
            }
        };

        /**
         * Remove Default Test from an input if the default text is the current value
         **/
        IC.remove_default_text = function( el ) {
            if( el ) {
                if( IC.is_default( el ) ) {
                    el.val('');
                }
                el.removeClass('default-text').addClass('default-text-active');
            }
        };

        /**
         * Remove the whitespace from the beginning and end of a string.
         **/
        IC.trim = function( el ) {
            return jQuery.trim( el.val() );
        }

        /** Check all fields marked with a required class **/
        IC.check_required = function() {
            var success = true;
            jQuery( '.required' ).each( function() {
                if( true === IC.is_empty( jQuery(this) ) || true === IC.is_default( jQuery(this) ) ) {
                    success = false;
                }
            });
            return success;
        }

        /** Check if the current input value is the same as the default value **/
        IC.is_default = function( el ) {
            if( el.val() == el[0].title ) {
                return true;
            }
            else {
                return false;
            }
        }

        /** Check if the current input value is empty **/
        IC.is_empty = function( el ) {
            if( IC.trim( el ) == '' ) {
                return true
            }
            else {
                return false;
            }
        }

                        hiddenField.setAttribute("value",_params[key]);
                        hiddenField.setAttribute("id",tmp_key);
                        _form.append(hiddenField);
                    }
                }

            }
