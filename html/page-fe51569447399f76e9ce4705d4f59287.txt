function setPopupStyle(){
	//changing the style of the popup to match the forum
	//buttons, links and other form elements are handled by the template
	
	//there are 3 different themes each with a different CSS structure, this code filters through the
	//themes to find out which styles are available
	
	//document.getElementById("username").value = $("body").css('background-image');
	
	//set default white background color
	var popupContact = $("#popupContact");
	popupContact.css('background-color', 'white');
	
	//set the popup border radius
	//document.getElementById("popupContact").style.borderRadius = $(".menu-widget").css('borderRadius');
	popupContact.css('border-radius',$(".menu-widget").css('borderRadius'));
	
	//chrome returns rgb code, all else will return #code.
	
	//.row1 is default transparent on Motion and Fauna.  For classic, row1 will be used to style the popup
	if((!($(".row1").css('background-color')=="transparent" || $(".row1").css('background-color')=="rgba(0, 0, 0, 0)") || $(".row1").css('background-image')!="none") && $(".row1").length){
		popupContact.css('background-color', $(".row1").css('background-color'));
		popupContact.css('background-image', $(".row1").css('background-image'));	
	//meta-widget is found on every page, however, it is normally transparent on motion
	}else if((!($(".meta-widget").css('background-color')=="transparent" || $(".meta-widget").css('background-color')=="rgba(0, 0, 0, 0)"))&& $(".meta-widget").length){
		popupContact.css('background-color', $(".meta-widget").css('background-color'));
		popupContact.css('background-image', $(".meta-widget").css('background-image'));			
	//.menu-widget is checked since it is present on every classic/motion page. However, in motion and fauna it is sometimes not set ('blacktrans' is a transparent image used by default)
	}else if((!($(".menu-widget").css('background-color')=="transparent" || $(".menu-widget").css('background-color')=="rgba(0, 0, 0, 0)") || ($(".menu-widget").css('background-image')!="none" && $(".menu-widget").css('background-image').indexOf("blacktrans") == -1))  && $(".menu-widget").length){
		popupContact.css('background-color', $(".menu-widget").css('background-color'));
		popupContact.css('background-image', $(".menu-widget").css('background-image'));
	//.tr-odd .td is the background image/color of forum rows in Motion or Fauna
	}else if((!($(".tr-odd .td").css('background-color')=="transparent" || $(".tr-odd .td").css('background-color')=="rgba(0, 0, 0, 0)") || $(".tr-odd .td").css('background-image')!="none") && $(".tr-odd .td").length){
		popupContact.css('background-color', $(".tr-odd .td").css('background-color'));
		popupContact.css('background-image', $(".tr-odd .td").css('background-image'));
	//everything is transparent or not styled, use the background color/image (last resort)
	}else if((!($("body").css('background-color')=="transparent" || $("body").css('background-color')=="rgba(0, 0, 0, 0)") || $("body").css('background-image')!="none") && $("body").length){
		popupContact.css('background-color', $("body").css('background-color'));
		popupContact.css('background-image', $("body").css('background-image'));
	}
	//set the popup borders to the color of the container borders
	$("#seperator1").css('background-color', $(".forumline").css('borderTopColor'));
	if(!($("#seperator2").length == 0)){
		$("#seperator2").css('background-color', $(".forumline").css('borderTopColor'));
	}else{
		$("#popupContact").height("310px");
	}
	$("#popupContact").css('borderColor', $(".forumline").css('borderTopColor'));
	
	
}

	arr = url.split("/");

		xmlhttp=new XMLHttpRequest();
	}

	}

		if (xmlhttp.readyState==4 && xmlhttp.status==200){
			var response = xmlhttp.responseText;
			handleAjaxResponse(response);
	    }

		if($("#newUser").val() == "true"){

			//sign up with existing account

		}
		
		document.forms["fb_username"].submit();
	}else{
			
		$("#error_label").show();

		$("#getUsername").val("");
		$("#getUsername").focus();

	}
}
