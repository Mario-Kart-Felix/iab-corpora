<script type="text/javascript">
  // If the user has closed, or the user is logged in, don't bother
  if (!$.cookie("nosignup") && !$.cookie("USER_ID"))
  {
    // Get the number of times the user has visited
    var counter = parseInt($.cookie("signup")) || 0;

    // If greater than this, popup the box
    if (counter > 2)
    {
       $('#sign_up').lightbox_me({
              centered: true,
              onLoad: function() {
                  $('#sign_up').find('input:first').focus()
                  }
              });
    }

    // Increase cookie counter by one
    $.cookie("signup", (counter + 1));
  }

  //  When the user clicks on the x, set the nosignup cookie for 90 days
  $('#close_x').click(function(e) {
    $.cookie("nosignup", "1", { expires: 90 });
  });


  $('#try-1').click(function(e) {
    $('#sign_up').lightbox_me({
        centered: true,
        onLoad: function() {
            $('#sign_up').find('input:first').focus()
            }
        });
    e.preventDefault();
});

</script>

	}
	
	document.location = u;
}
