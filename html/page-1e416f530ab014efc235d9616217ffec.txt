function publicSendMessage(data)
{
	_gaq.push(['_trackEvent', "product_detail", data.type+'_open', data.target_id]);
	var public_send_message_dialog_form = $("#jtmpl_dialog_public_send_message").tmpl(data);
	var dialog_form = public_send_message_dialog_form.find("form[name=send_message]");
	
	dialog_form.validate({
		submitHandler: function(form){
			var message = dialog_form.serialize();
			var email = dialog_form.find("input[name=\"message[email]\"]").val();
			var text = dialog_form.find("textarea[name='message[message]']").val();
			var reg = /http:\/\/[\w-]*(\.[\w-]*)+/ig;
			var reg_email = /^[A-z0-9-_\.]+@[A-z0-9-_\.]+$/i;
			var var_arr = new Array('.com', '.cn', '.hk', '.org', '.net', '.in', '.mobi', '.gov', '.info', '.biz', '.cc', '.tv', '.asia', '.tw', '.name', '.me', '.us', '.ca');
			if(reg.test(text) || inText(text.toLowerCase(), var_arr) || reg_email.test(text)){
				showCaution('Sorry! Only registered users can send URL or Email addresses in inquiry. Please register/login to send this message.', null, 'pop', null, false);
				return false;
			}

			var working = showWorking(null, 'Sending...');
			$.post("public/messages/send/", message, function(back_data){
				working.liteDialog('hide');
				back_data = $.parseJSON(back_data);
				if(typeof(back_data) == 'string'){
					showCaution(back_data);
					return;
				}
				if(back_data > 0)
				{
					if(!public_user_id)
					{
						var pageTracker = _gat._getTracker("UA-11204542-2");
						pageTracker._initData();
						pageTracker._trackPageview('/public/user/confirm_email');
					}
					
					public_user_id = back_data;
					public_send_message_dialog_form.liteDialog('hide');
//					savePublicUserActions("public_" + data.type +"_inquiries", null, data.to, data.target_id);
					showNotice('Your message has been sent.', null, 'pop', null, true);
				}
				else if(back_data == -1)
				{
					public_send_message_dialog_form.liteDialog('hide');
					showCaution("Sorry, You have reached the sending limit for today. <a href='user/vip'>Become our premium member</a> to send more, or comeback tomorrow. Thank you for your support!", null, 'pop');
				}
				else
				{
					showCaution('Message send failed, please try again', null, 'pop', null, true);
				}
			});
		}
	});
	
	public_send_message_dialog_form.liteDialog({
		width: '790px'
	});
	
	public_send_message_dialog_form.find('#f_cancel').click(function(e){
		e.preventDefault();
		public_send_message_dialog_form.liteDialog('hide');	
	});

		e.preventDefault();
		if(dialog_form.find("input[name=\"message[send_to_similar_product]\"]").attr('checked') == true)
		{
			var send_to_similar_product = 1;
		}
		else
		{
			var send_to_similar_product = 0;
		}
		_gaq.push(['_trackEvent', "product_detail", data.type+'_send', data.target_id, send_to_similar_product]);
		if(!public_user_id) _gaq.push(['_trackEvent', 'register', 'register ' + "product detail" + ' inquire success']);
		dialog_form.submit(); // Trigger Validation
	});

			public_send_message_dialog_form.find("#f_description").text(summary);

			dialog_form.submit();
		}
	});
