function updtchk(chid,id){

	// This manages the collection of vehicles that the
	// user has selected for comparison

	var mychk = document.getElementById((chid+id));
	var nbrselected = 0;

	// build a string containing all of the previously selected records (ids)
	var existingids = "";
	if (document.getElementById("cid1").value.length > 2) {
		existingids = existingids+document.getElementById("cid1").value +"|";
		nbrselected = nbrselected + 1;
	}
	if (document.getElementById("cid2").value.length > 2) {
		existingids = existingids+document.getElementById("cid2").value +"|";
		nbrselected = nbrselected + 1;
	}
	if (document.getElementById("cid3").value.length > 2) {
		existingids = existingids+document.getElementById("cid3").value +"|";
		nbrselected = nbrselected + 1;
	}
	if (document.getElementById("cid4").value.length > 2) {
		existingids = existingids+document.getElementById("cid4").value +"|";
		nbrselected = nbrselected + 1;
	}

	if (nbrselected == 4 && mychk.checked) {
		alert("Maximum number (4) vehicles have been selected.");
	}

	if (nbrselected > 3 && mychk.checked) {
		document.getElementById("cbid1"+id).checked = false;
		document.getElementById("cbid2"+id).checked = false;
		id = "";
	}

	// If this vehicle is already in the list, remove it...
	// otherwise append it to the list
	var nowstring ="";
	if (existingids.indexOf(id) > -1) {
		nowstring = existingids.replace(id,"");
	} else if (mychk.checked){
		nowstring = existingids+id+"|";
	}

	// If the user has unchecked the vehicle, remove it from the list
	if (mychk.checked == false) {
		nowstring = existingids.replace(id,"");
	}

	if (nowstring.charAt(0) == "\|") {
		nowstring = nowstring.substr(1);
	}

	var selids = nowstring;
	var selids1 = selids.replace('||||','|');
	var selids2 = selids1.replace('|||','|');
	var selids3 = selids2.replace('||','|');
	selids = selids3.substring(0,selids3.length-1);

	// convert the string of selected ids into an array of selected ids
	MakeArry = selids.split("\|");

	// Clear out previous variables	
	document.getElementById("cid1").value = "";
	document.getElementById("cid2").value = "";
	document.getElementById("cid3").value = "";
	document.getElementById("cid4").value = "";

	if (MakeArry.length > 0) {
		document.getElementById("cid1").value = MakeArry[0];
	}
	if (MakeArry.length > 1) {
		document.getElementById("cid2").value = MakeArry[1];
	}
	if (MakeArry.length > 2) {
		document.getElementById("cid3").value = MakeArry[2];
	}
	if (MakeArry.length > 3) {
		document.getElementById("cid4").value = MakeArry[3];
	}

	// check ids on other tab.
	if (mychk.checked){
		if (chid == "cbid1") {
			var mychk2 = document.getElementById(("cbid2"+id));
			mychk2.checked = true;
		} 

		if (chid == "cbid2") {
			var mychk2 = document.getElementById(("cbid1"+id));
			mychk2.checked = true;
		} 
	} else {
		document.getElementById(("cbid1"+id)).checked = false;
		document.getElementById(("cbid2"+id)).checked = false;
	}

}

		}

			paramstr=paramstr+"&pageno="+mypagenbr;

		} else {
			paramstr = paramstr +"&"+testurl[x];
		}

	}

	if (!pagefound) {paramstr = paramstr +"&pageno="+mypagenbr;}
	if (!sortfound) {paramstr = paramstr +"&sortBy=Comb";}
	if (!tabfound) {paramstr = paramstr +"&tabView="+mytabnbr;}
