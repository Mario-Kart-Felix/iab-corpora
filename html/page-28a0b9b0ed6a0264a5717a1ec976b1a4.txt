	jQuery(document).on('leadform:zipchange', function(event, selectedProgram) {
		var $zipcode = jQuery('#zipcode');
		if ($zipcode.val()=='' || $zipcode.val()==lastzip) {
			return;
		}
		
		jQuery(document).trigger('leadform:ziplookup');
		
		var program_type = '';
		var $inputs = jQuery('input[name="program_type"]');
		if ($inputs.length > 0) {
			if ($inputs.eq(0).attr('type') == 'radio') {
				program_type = '&program_type=' + $inputs.filter(':checked').val();
			} else {
				program_type = '&program_type=' + $inputs.eq(0).val();
			}
		}
		
		lastzip = $zipcode.val();
		jQuery.ajax({
			dataType: 'json',
			url: '/themes/_leadform/ziplookup.php?zipcode=' + $zipcode.val() + program_type,
			success: function(json) {
				$programs = jQuery('#program');
				if (json.programs && $programs.length>0 && $programs.prop('tagName')=='SELECT') {
					jQuery(document).trigger('leadform:programupdate');
					
					var count = 0;
					for (var i in json.programs) {
						if (typeof(json.programs[i]) == 'function') {
							continue;
						}
						count++;
					}
					
					$programs.empty();
					if (count > 0) {
						$programs.append(jQuery('<option></option>').val('').text('-- Select a Program --'));
						if (jQuery('#custom_programs').length>0 && jQuery('#custom_programs').val()!='') {
							var custom_programs = jQuery('#custom_programs').val().toUpperCase().split(/[^\w]+/);
							for (var i in json.programs) {
								if (i.match(/OL$/)) {
									continue;
								}
								if (custom_programs.indexOf(i.toUpperCase()) == -1) {
									delete json.programs[i];
								}
							}
						}
						
						for (var i in json.programs) {
							$programs.append(jQuery('<option></option>').val(i).text(json.programs[i]));
						}
						
						if (selectedProgram) {
							$programs.val(selectedProgram);
						}
						
						if (json.online == true) {
							jQuery(document).trigger('leadform:onlinenotice');
						} else {
							jQuery(document).trigger('leadform:groundnotice');
						}
					} else {
						$programs.append(jQuery('<option></option>').val('').text('No Programs in Your Area'));
					}
					
					jQuery(document).trigger('leadform:programupdated');
				}
				
				if (json.location) {
					if (jQuery('#city') && json.location.city) {
						jQuery('#city').val(json.location.city);
						jQuery(document).trigger('leadform:city-update');
					}
					
					if (jQuery('#state').length>0 && json.location.state) {
						jQuery('#state').val(json.location.state);
						jQuery(document).trigger('leadform:state-update');
					}
					
					if (jQuery('#country') && json.location.country) {
						jQuery('#country').val(json.location.country);
						jQuery('leadform:country-update');
					}
				}
				
				jQuery(document).trigger('leadform:endziplookup');
			}
		});
	});
