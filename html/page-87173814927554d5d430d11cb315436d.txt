<p>Motorcycles, Scooters, and Mopeds.</p>

// This function will set the startup currency according to the user's country iso
function setCurrency(currentCurrency)
{
	$.cookie("currency-code", currentCurrency,{ expires: 7, path: '/', domain: '.reportbuyer.com'});
	
	$('.curr a').removeClass('hi');
	$('a[rel='+currentCurrency+']').addClass('hi');
	$('.product-pricing span.price').css('display', 'none');
	$('.product-pricing span.'+currentCurrency).css('display', 'block');
	
	$('#ajax-product-pricing .price').not('#ajax-product-pricing .striken').each(function()
	{
		$(this).html($(this).text());
	});
	
	var price 		= $('#ajax-product-pricing a:first .'+currentCurrency+':not(.striken)').text().substring(1);
	var symbol 		= $('#ajax-product-pricing a:first .'+currentCurrency+':not(.striken)').text().substring(0, 1);
	var symbol_html	= unescape('%3Ci itemprop="currency" content="'+currentCurrency+'"%3E') + symbol + unescape('%3C/i%3E');
	var price_html  = unescape('%3Ci itemprop="price"%3E') + price + unescape('%3C/i%3E');
	
	$('#ajax-product-pricing a:first .'+currentCurrency+':not(.striken)').html(symbol_html + price_html);
}

// will assign ajax function to form submits which will
// fetch itself and then grab the form response from the 
// page and display it
//
// @param container id of container div
// @param google_event_id google event id
// @param google_event_name google event name
function product_enquiry (container, google_event_id, google_event_name)
{
	$(container + " form").submit(function()
	{
		var post = $(this).serialize();
		
		// add loading
		$(this).find(".submit").html(unescape("%3Cimg src='/images/ajax-loader.gif' /%3E"));
		
		$.ajax(
		{
			type: "POST",
			data: post,
			timeout: 9000,
			url: window.location.href,
			context: document.body,
			success: function(data)
			{
				$(container + " div:first").html( $(data).find(container + " div:first").html() );
	
				// resize container
				$.colorbox.resize();

				// reload the captcha to fix session problem
				var captcha = $(container + " .captcha-img img");
				captcha.attr("src", captcha.attr("src") + "&random=" + Math.floor(Math.random()*11));

				// set up functions again
				product_enquiry(container, google_event_id, google_event_name);

				// check if validate if has trigger google event
				if ($(container + " .warning").length == 0)
				{
					_gaq.push(['_trackEvent', google_event_id, google_event_name, get_product_code(), get_price()]);
					//pageTracker._trackEvent(google_event_id, google_event_name, get_product_code(), get_price());
				}
			},
			error: function () 
			{
        		alert('Sorry there was a problem sending your enquiry. Please reload the page and try again.');
   			}

		});
		
		return false;
	});
}
