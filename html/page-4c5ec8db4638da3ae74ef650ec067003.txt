    function FixMake(strMake) {

        var strNewMake = '';

        strNewMake = strMake.replace(/Chevrolet Truck, Van, SUV/i, 'Chevy');
        strNewMake = strNewMake.replace(/Chevrolet/i, 'Chevy');
        strNewMake = strNewMake.replace(/Dodge Truck, Van, SUV/i, 'Dodge');
        strNewMake = strNewMake.replace(/Ford Truck, Van, SUV/i, 'Ford');
        strNewMake = strNewMake.replace(/GMC Truck, Van, SUV/i, 'GMC');
        strNewMake = strNewMake.replace(/Mazda Truck, Van, SUV/i, 'Mazda');
        strNewMake = strNewMake.replace(/Volkswagen/i, 'VW');
        strNewMake = strNewMake.replace(/Toyota Truck, Van, SUV/i, 'Toyota');
        strNewMake = strNewMake.replace(/Oldsmobile/i, 'Olds');
        strNewMake = strNewMake.replace(/Nissan Truck, Van, SUV/i, 'Nissan');

        return strNewMake;
    }
