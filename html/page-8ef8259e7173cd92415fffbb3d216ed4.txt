    function notifyGigyaLogin()
    {
        var GigyaUID    = Gigya.UID || "";
            new_user    = (GigyaUID === "") ? true : false,
            timestamp   = Math.floor((new Date).getTime()/1000);

        // Notify Gigya that the user has logged in 
        // from Clickshare
        $.ajax({
            url: '/section/_GigyaSignature',
            type: 'get',
            data: { base_string : CSUser.userID, timestamp: timestamp },
            success: function(res) {
                signature = res;
                $(document).trigger('SignatureCreated');
            }
        });

        $(document).unbind('SignatureCreated');
        $(document).bind('SignatureCreated', function(){ 
            
            var params       = {
                    siteUID      : CSUser.userID,
                    UIDTimestamp : timestamp,
                    UIDSig       : signature,
                    newUser      : new_user,
                    userInfo     : {
                        firstName       : CSUser.firstName,
                        lastName        : CSUser.lastName,
                        thumbnailURL    : CSUser.thumbnailURL,
                        email           : CSUser.email
                    }
            };
        
            gigya.socialize.notifyLogin(params);
            
        });

    }

    var readCookie = function (name) 
    {
        var nameEQ = name + "=";
        var ca = document.cookie.split(';');
        for (var i = 0; i < ca.length; i++) {
            var c = ca[i];
            while (c.charAt(0) == ' ') c = c.substring(1, c.length);
            if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
        }
        return null;
    }
