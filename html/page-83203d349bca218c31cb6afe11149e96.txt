(function($){
	/* hoverIntent by Brian Cherne */
	$.fn.hoverIntent = function(f,g) {
		// default configuration options
		var cfg = {
			sensitivity: 7,
			interval: 100,
			timeout: 0
		};
		// override configuration options with user supplied object
		cfg = $.extend(cfg, g ? { over: f, out: g } : f );

		// instantiate variables
		// cX, cY = current X and Y position of mouse, updated by mousemove event
		// pX, pY = previous X and Y position of mouse, set by mouseover and polling interval
		var cX, cY, pX, pY;

		// A private function for getting mouse position
		var track = function(ev) {
			cX = ev.pageX;
			cY = ev.pageY;
		};

		// A private function for comparing current and previous mouse position
		var compare = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			// compare mouse positions to see if they've crossed the threshold
			if ( ( Math.abs(pX-cX) + Math.abs(pY-cY) ) < cfg.sensitivity ) {
				$(ob).unbind("mousemove",track);
				// set hoverIntent state to true (so mouseOut can be called)
				ob.hoverIntent_s = 1;
				return cfg.over.apply(ob,[ev]);
			} else {
				// set previous coordinates for next time
				pX = cX; pY = cY;
				// use self-calling timeout, guarantees intervals are spaced out properly (avoids JavaScript timer bugs)
				ob.hoverIntent_t = setTimeout( function(){compare(ev, ob);} , cfg.interval );
			}
		};

		// A private function for delaying the mouseOut function
		var delay = function(ev,ob) {
			ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t);
			ob.hoverIntent_s = 0;
			return cfg.out.apply(ob,[ev]);
		};

		// A private function for handling mouse 'hovering'
		var handleHover = function(e) {
			// next three lines copied from jQuery.hover, ignore children onMouseOver/onMouseOut
			var p = (e.type == "mouseover" ? e.fromElement : e.toElement) || e.relatedTarget;
			while ( p && p != this ) { try { p = p.parentNode; } catch(e) { p = this; } }
			if ( p == this ) { return false; }

			// copy objects to be passed into t (required for event object to be passed in IE)
			var ev = jQuery.extend({},e);
			var ob = this;

			// cancel hoverIntent timer if it exists
			if (ob.hoverIntent_t) { ob.hoverIntent_t = clearTimeout(ob.hoverIntent_t); }

			// else e.type == "onmouseover"
			if (e.type == "mouseover") {
				// set "previous" X and Y position based on initial entry point
				pX = ev.pageX; pY = ev.pageY;
				// update "current" X and Y position based on mousemove
				$(ob).bind("mousemove",track);
				// start polling interval (self-calling timeout) to compare mouse coordinates over time
				if (ob.hoverIntent_s != 1) { ob.hoverIntent_t = setTimeout( function(){compare(ev,ob);} , cfg.interval );}

			// else e.type == "onmouseout"
			} else {
				// unbind expensive mousemove event
				$(ob).unbind("mousemove",track);
				// if hoverIntent state is true, then call the mouseOut function after the specified delay
				if (ob.hoverIntent_s == 1) { ob.hoverIntent_t = setTimeout( function(){delay(ev,ob);} , cfg.timeout );}
			}
		};

		// bind the function to the two event listeners
		return this.mouseover(handleHover).mouseout(handleHover);
	};
	
})(jQuery);

<span class="" id="result_box" lang="en"><span class="hps"><br /></span></span></div>
<div style="text-align: justify;">
<span class="" id="result_box" lang="en"><span class="hps">South Korean</span> <span class="hps">car manufacturer</span><span>,</span> <span class="hps">Hyundai</span> <span class="hps">Motor</span> <span class="hps">Co</span><span>,</span> <span class="hps">launched a prototype of</span> <span class="hps">a </span></span><span class="" id="result_box" lang="en"><a href="http://automotive-innovations.blogspot.com/2012/09/hyundai-launches-i30-three-door-hatchback" target="_blank"><b><span class="hps"><span class="" id="result_box" lang="en"><span class="hps">i30</span> <span class="hps alt-edited"></span></span>three-door</span> </b></a><span class="hps"><a href="http://automotive-innovations.blogspot.com/2012/09/hyundai-launches-i30-three-door-hatchback" target="_blank"><b>hatchback</b></a> model</span> <span class="hps alt-edited">in the event of</span> <span class="hps">Paris</span> <span class="hps">Motorshow</span><span>,</span> <span class="hps">29 September 2012</span><span>.</span> <span class="hps">Sedan</span> <span class="hps">stump</span> <span class="hps">is likely</span> <span class="hps">assembled</span> <span class="hps">and marketed</span> <span class="hps">in Europe</span> <span class="hps">by the end of</span> <span class="hps">2013.</span></span></div>
<div style="text-align: justify;">
<br /></div>
<div style="text-align: justify;">
<span class="" id="result_box" lang="en"><span class="hps">Hyundai</span> <span class="hps">launches</span> <span class="hps">i30</span> <span class="hps">Three</span> <span class="hps">Doors</span> <span class="hps">week after</span> <span class="hps">compatriot</span> <span class="hps">manufacturer</span><span>,</span> <span class="hps">Kia</span><span>, has announced</span> <span class="hps">a marketing plan</span> <span class="hps">Pro_ceed</span> <span class="hps">three-door hatchback</span><span>.</span> <span class="hps">In a</span> <span class="hps">statement</span><span class="">,</span> <span class="hps">Hyundai</span> <span class="hps">said the</span> <span class="hps">i30</span> <span class="hps">Three</span> <span class="hps">Doors</span> <span class="hps">will compete</span> <span class="hps">in the</span> <span class="hps atn">semi-</span><span class="">premium</span> <span class="hps">sedan</span> <span class="hps">segment</span> <span class="hps">and is designed specifically</span> <span class="hps">for the European market</span><span class="">.</span></span></div>
<div style="text-align: justify;">
<br /></div>
<div style="text-align: justify;">
<span class="" id="result_box" lang="en"><span class=""><span class="" id="result_box" lang="en"><span class="hps">"Designed</span><span class="">, manufactured</span><span>,</span> <span class="hps">and marketed</span> <span class="hps">in Europe</span><span>,</span> <span class="hps">for</span> <span class="hps alt-edited">European society</span><span>,</span><span>"</span> <span class="hps">Hyundai</span> <span class="hps">said in a statement</span> <span class="hps">quoted</span> <span class="hps">by autoevolution</span><span>,</span> <span class="hps">Monday, September 17, 2012</span><span class="">.</span></span></span></span></div>
<div style="text-align: justify;">
<br /></div>
<div style="text-align: justify;">
<span class="" id="result_box" lang="en"><span class=""><span class="" id="result_box" lang="en"><span class=""><span class="" id="result_box" lang="en"><span class="hps">Unlike</span> <span class="hps">other</span> <span class="hps">hatchback</span><span>,</span> <span class="hps">Hyundai</span> <span class="hps">applied</span> <span class="hps">design</span> <span class="hps">buttocks</span> <span class="hps">upright</span> <span class="hps">or</span> <span class="hps">A</span><span class="atn">-</span><span>Pillar</span> <span class="hps">for the</span> <span class="hps">i30</span><span>.</span> <span class="hps">Its appearance is</span> <span class="hps">enhanced</span> <span class="hps">by the</span> <span class="hps">latest model</span> <span class="hps">grille</span> <span class="hps">and</span> <span class="hps">bumper</span> <span class="hps">and the rear wing</span> <span class="hps">are</span> <span class="hps">more muscular</span><span>.</span></span></span></span></span></span></div>
