	currentURL = theURL;

	var currentHref = $(leftLinks[i]).attr("href");

		$(leftLinks[i]).parent().addClass("current");
		break;

function checkListAgainstHref(href, domainList)
{
	for(var i = 0; i < domainList.length; i++)
	{
		//Use values from the array to match against the href value
		if(href.search(domainList[i]) > -1)
		{
			//if the one found is the current domain, don't add
			var currentDomain = window.location.href;
			if(currentDomain.search(domainList[i]) > -1)
			{
				return false;
			}
			else
			{
				return true;
			}
		}
	}
	return false;
};


function addGACodeToATags(domainList)
{
	//Get all <a> tags on the page. If the destination domain is in the domain list, add the onClick event
	var aTags = document.getElementsByTagName('a');
	var thisHref = "";
	var currentOnClick = "";
	var aTagGACode = "newTracker._link(this.href); return false;";
	for(var i = 0; i < aTags.length; i++)
	{
		thisHref = aTags[i].href;
		if(checkListAgainstHref(thisHref, domainList) == true)
		{
			//First, get the current onClick event
			currentOnClick = (aTags[i].getAttribute('onClick'));
			if(currentOnClick != null && (currentOnClick.toLowerCase()).search("return") > -1)
			{
				//currently existing return function, this is going to be weird. For now, just append to the end of the string
				currentOnClick += aTagGACode;
			}
			else
			{
				//no currently existing return
				currentOnClick = aTagGACode;
			}
			
			aTags[i].setAttribute('onClick', currentOnClick);
		}
	}
};

			//Add the GA code to the onSubmit

			}
			else
			{
				currentOnSubmit = formTagGACode;
			}
