                if (data.status == "success") {
                    $('#invitationBox').val('Email');

                    $('#Success').fadeIn('fast');
                    $('#RequestInvitation').hide();

                    $('#RequestInvitation').hide();

            }

    }

<script>
function ContainsItem(itemToFind, allItems)
{
    var isFound = false;
    
    for (var i = 0; i < allItems.length; i++)
    {
        if (itemToFind.substr(0,allItems[i].length)==allItems[i]) //match prefixes
        {
            isFound = true;
            break;
        }
    }
                
    return isFound;
}

function InitializeMarketo()
{
    if (typeof Munchkin !== 'undefined') // Might be undefined if munchkin.js script hasn't finished loading
    {
        var marketoAccountId = '417-NLK-080'; // '417-NLK-080';
        var pagePath = window.location.pathname.toLowerCase();
        var pagesToTrack = [
            '/hr/',
            '/pro/',
            '/biz/',
            '/compensation',
            '/payscale-index'
            ];

        if (window.location.hostname.toLowerCase() == 'marketrate.payscale.com' || ContainsItem(pagePath, pagesToTrack))
        {
            Munchkin.init(marketoAccountId);
        }
        else
        {
            // Workaround for Marketo not being able to handle all of our traffic
            Munchkin.init(marketoAccountId, {cookieAnon: false});
        }
    }
}
InitializeMarketo();
