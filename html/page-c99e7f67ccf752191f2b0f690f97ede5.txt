            function atl_load_year_memorize() {
                obj_value = document.getElementById('atl_hidden_class_y').defaultValue;
                
                var array_years = new Array();
                var array_years_add = new Array();

                jQuery('select[name="class_year"] option').each(function(){
                    if(jQuery(this).attr('value') != '') {
                        array_years[j] = jQuery(this).attr('value');
                        j++;
                    }
                });
                
                jQuery('select[name="class_year"] option').each(function(){
                    if(jQuery(this).attr('value') != '') $(this).remove();
                });
                
                for (var i = 0; i < array_years.length; i++) {
                    if (jQuery.inArray(array_years[i], array_years_add) == -1) {
                        $('select[name="class_year"]').append($('<option ' + (array_years[i] == obj_value ? 'selected' : '') + '></option>').attr('value', array_years[i]).text((array_years[i] == -1 ? 'All years' : array_years[i])));
                        array_years_add[i] = array_years[i];
                    }
                }
                
                atl_load_mark_by_year_memorize();
            }
