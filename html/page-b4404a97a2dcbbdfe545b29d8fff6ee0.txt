function PutDropOffDate(pickup_date)
{
    if(trim(document.getElementById('drop_off_date').value) == "")
    {
        var supp_type = document.getElementById('supplier_cat').value;
        var pickup_date = document.getElementById('pick_up_date').value
        var pickup_date_arr = pickup_date.split(' ');
        if(pickup_date_arr.length == 3)
        {
            var day = pickup_date_arr[0];
            if(day == "08")
                day = "8";
            if(supp_type == "car_leasing")
                day = parseInt(day)+21;
            else
                day = parseInt(day)+7;
                    
            var month = getMonthNo(pickup_date_arr[1]);
            var year = pickup_date_arr[2];
            var pick_date = new Date();
            pick_date.setFullYear(year,month,day);
            
            var months = new Array('January','February','March','April','May','June','July','August','September','October','November','December');
            var drop_date =  pick_date.getDate() + " " + months[pick_date.getMonth()] + " " + pick_date.getFullYear();
            document.getElementById('drop_off_date').value = drop_date;
        }
    }
}

function trim(s) 
{
  while (s.substring(0,1) == ' ') 
  {
		s = s.substring(1,s.length);
  }
  while (s.substring(s.length-1,s.length) == ' ') 
  {
		s = s.substring(0,s.length-1);
  }
  return s;
}

function getMonthNo(strMonth)
{
    var month_no = "0";
    
    if(strMonth == "January")
        month_no = "0";
    else if(strMonth == "February")
        month_no = "1";
    else if(strMonth == "March")
        month_no = "2";
    else if(strMonth == "April")
        month_no = "3";
    else if(strMonth == "May")
        month_no = "4";
    else if(strMonth == "June")
        month_no = "5";
    else if(strMonth == "July")
        month_no = "6";
    else if(strMonth == "August")
        month_no = "7";
    else if(strMonth == "September")
        month_no = "8";
    else if(strMonth == "October")
        month_no = "9";
    else if(strMonth == "November")
        month_no = "10";
    else if(strMonth == "December")
        month_no = "11";
    return month_no;                                            
}

function CheckForm()
{
    var is_valid = true;
    var error_messages = "Please Give OR Correct The Following Detail(s) As Your Search Criteria:\n\n";
    var focus_on = "";
    var currency_list = document.getElementById('currency_list').value;
    var pick_up_country = document.getElementById('pick_up_country').value;
    var drop_off_country = document.getElementById('drop_off_country').value;
    var pick_up_city = document.getElementById('pick_up_city').value;
    var drop_off_city = document.getElementById('drop_off_city').value;
    var pick_up_date = document.getElementById('pick_up_date').value;
    var drop_off_date = document.getElementById('drop_off_date').value;
    
    if(currency_list.search(/\S/) == -1 )
	{
	   error_messages += "* Please Select Your Currency.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "currency_list";
	}
	if(pick_up_country.search(/\S/) == -1 )
	{
	   error_messages += "* Please Select Pick Up Country.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "pick_up_country";
	}
	if(drop_off_country.search(/\S/) == -1 )
	{
	   error_messages += "* Please Select Drop Off Country.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "drop_off_country";
	}
	if(pick_up_city.search(/\S/) == -1 )
	{
	   error_messages += "* Please Select Pick Up City.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "pick_up_city";
	}
	if(drop_off_city.search(/\S/) == -1 )
	{
	   error_messages += "* Please Select Drop Off City.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "drop_off_city";
	}
	if(pick_up_date.search(/\S/) == -1 )
	{
	   error_messages += "* Please Enter Your Pick Up Date.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "pick_up_date";
	}
	
    var pickup_date = document.getElementById('pick_up_date').value
    if(pickup_date != "")
    {
        var pickup_date_arr = pickup_date.split(' ');
        if(pickup_date_arr.length != 3)
        {
           error_messages += "* Pick Up Date Must In Format Like (08 January 2008).\n";
           is_valid = false;
           if(focus_on == "")
                focus_on = "pick_up_date";
        }
        else
        {
            var day = pickup_date_arr[0];
            var month = getMonthNo(pickup_date_arr[1]);
            var year = pickup_date_arr[2];
            
            var pick_date = new Date();
            pick_date.setFullYear(year,month,day);
            var today = new Date();
            if (today>pick_date)
            {
               error_messages += "* Pick Up Date Can Not Be Previous Date.\n";
               is_valid = false;
               if(focus_on == "")
                    focus_on = "pick_up_date";        
            }
         }
    }
        
	if(drop_off_date.search(/\S/) == -1 )
	{
	   error_messages += "* Please Enter Your Drop Off Date.\n";
	   is_valid = false;
	   if(focus_on == "")
	        focus_on = "drop_off_date";
	}
	
    var dropoff_date = document.getElementById('drop_off_date').value
    if(dropoff_date != "")
    {
        var pickup_date_arr = pickup_date.split(' ');
        var pday = pickup_date_arr[0];
        var pmonth = getMonthNo(pickup_date_arr[1]);
        var pyear = pickup_date_arr[2];
        
        var dropoff_date_arr = dropoff_date.split(' ');
        if(dropoff_date_arr.length != 3)
        {
           error_messages += "* Drop Off Date Must In Format Like (08 January 2008).\n";
           is_valid = false;
           if(focus_on == "")
                focus_on = "drop_off_date";
        }
        else
        {
            var dday = dropoff_date_arr[0];
            var dmonth = getMonthNo(dropoff_date_arr[1]);
            var dyear = dropoff_date_arr[2];
            
            var pick_date = new Date();
            pick_date.setFullYear(pyear,pmonth,pday);
            var drop_date = new Date();
            drop_date.setFullYear(dyear,dmonth,dday);
            
            if (pick_date>drop_date)
            {
               error_messages += "* Drop-Off Date Can Not Be Less Than Pick-Up Date.\n";
	           is_valid = false;
	           if(focus_on == "")
	                focus_on = "drop_off_date";
            }
         }   
     }   
	
	if(is_valid == false)
	{
	    alert(error_messages);
	    document.getElementById(focus_on).focus();
	    return false;  
	}
	else
	{
        document.getElementById('topbar_searching').style.display = "";
        staticbar(); 
        document.getElementById('loading_page').style.display = "";
        document.getElementById('whole_page').style.display = "none";
        document.getElementById('body_tag').style.backgroundColor = "#99CCFF";
        var supp_type = document.getElementById('supplier_cat').value;
        var car_type = "";
        if(supp_type == "luxury_car")
            car_type = "Luxury Cars";
        else if(supp_type == "motor_bike")
            car_type = "Motor Bikes";
        else
            car_type = "Cars";
        document.getElementById('searching_what').innerHTML = car_type;
        return true;
	}
}
