{
    this.video = video;
    this.playerId = playerId;
    this.width = width;
    this.height = height;
    this.autoPlay = autoPlay;

        if (!this.runNow()) {
            var self = this;
            this.timer = setInterval(function() { self.runNow(); }, 1000);
        }

        if (typeof xml_url != 'undefined') {
            if (this.timer)
                clearInterval(this.timer);

            return true;
        }
        return false;
    }
}
