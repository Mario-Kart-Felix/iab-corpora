 * param string div_id  the id of the jonblock
 * returns void
 */
function reposition_jonbox(div_id)
{
	var jonbox = document.getElementById(div_id);

	var scroll_offset_x, scroll_offset_y, vp_width, vp_height;

	// Compute scroll offsets for different browsers
	if (self.pageYOffset)
	{
		scroll_offset_x = self.pageXOffset;
		scroll_offset_y = self.pageYOffset;
	}
	else if (document.documentElement && document.documentElement.scrollTop)
	{
		scroll_offset_x = document.documentElement.scrollLeft;
		scroll_offset_y = document.documentElement.scrollTop;

	else if (document.body)

		scroll_offset_x = document.body.scrollLeft;
		scroll_offset_y = document.body.scrollTop;

		vp_height = self.innerHeight;

/**
 * Fades in an object using a starting opacity
 * (In the jonbox namespace to avoid overriding)
 * param string obj_id  the id of the object
 * param int opacity  the starting opacity in percent (0-100)
 * returns void
 */
function jonbox_fade_in(obj_id, opacity)
{
	if (opacity > 100) { return; }

	// Set the opacity
	var obj = document.getElementById(obj_id);
	jonbox_set_opacity(obj, opacity);

	// Recurse
	window.setTimeout("jonbox_fade_in('" + obj_id + "', " + (opacity + 10) + ");", 5);
}

/**
 * Sets the opacity of an object
 * (In the jonbox namespace to avoid overriding)
 * param object obj  the object
 * param int opacity  the opacity in percent (0-100)
 */
function jonbox_set_opacity(obj, opacity)
{
	// Keep opacity within range
	if (opacity < 0)
	{
		opacity = 0;
	}
	else if (opacity > 100)
	{
		opacity = 100;
	}

	// Express the opacity as a ratio
	ratio = opacity / 100;

	// Cross-browser transparency
	obj.style.opacity = ratio;
	obj.style.MozOpacity = ratio;
	obj.style.KhtmlOpacity = ratio;
	obj.style.filter = "alpha(opacity=" + opacity + ")";
}
