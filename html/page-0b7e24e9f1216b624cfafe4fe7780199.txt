	function _submitValid()
	{
		var empty = true;
		var provinceCode = "KLMNP";
		var sessionProvince="ON"
		
		_errorMsgClose();
		
		var obj = null;
		if (document.getElementById('_postalCodeForm').style.display=='block'){
			obj = document.getElementById("_postalCode2");
		}else{
			obj = document.getElementById("a_postalCode2");
		}
		
		var pc = obj.value;
		
		pc=trimSpaces(pc).toUpperCase();
		
		inputPostalCode = pc;

		if(pc!="" && postcodevalidate("ca",pc))	{
			var sameSessionProvince = provinceEqual(sessionProvince,pc);
			var sameShoppingProvince = provinceEqual(shoppingProvince,pc);
  			if(sameSessionProvince && sameShoppingProvince){ //added wireless province
  				return true;
 			}else{
				
				if(!sameShoppingProvince && !empty){
					_boxServiceCheck1Close();
	 				showAddressNotEqualBox();
					return false;
				}
				if(!sameSessionProvince && empty){
	 				document.getElementById('_boxServiceCheck4').style.display='block'; 
					return false;
				}
				
 				return true;
 			}	 					
	 	}
	 	else{	 	
		 	if (pc==""){	 	
			 	document.getElementById('_boxServiceCheck3').style.display='block'; 
			 	
		  	}else {
		  		document.getElementById('_boxServiceCheck2').style.display='block'; 
		  	}
		 	
		 	obj.value='';
		 	obj.focus();
		  	return false;
	 	}
	
	}

	function _cableAddressCheck(){
		var retu = false;
		if(_submitValid()){

			var checkPackage=false;
			var inBrowseFlow=false;

			if(currLineCode=="H" && (currTargetUrl.indexOf("hpBrowseFlow")!=-1 ))
				inBrowseFlow=true;

			if(currLineCode=="H" && (currTargetUrl.indexOf("HPH_ORDER")!=-1 || currTargetUrl.indexOf("hpOrderPromotion")!=-1) && "false" == "true")//check HP package
				checkPackage=true;

			if(currLineCode=="H" && currTargetUrl.indexOf("hprecotool")!=-1 && "false" == "true")//check HP reco tool 
				checkPackage=true;

			if(currLineCode=="I" && (currTargetUrl.indexOf("ipsBuyFlowBegin")!=-1 || currTargetUrl.indexOf("ipsOrderPromotion")!=-1 || currTargetUrl.indexOf("ipsCheckServiceabilityByPostalCode")!=-1) && "false" == "true")//check IPS package
				checkPackage=true;
			
			if(currLineCode=="I" && currTargetUrl.indexOf("dialUpServiceBegin")!=-1 && "false" == "true")//check IPS package
				checkPackage=true;	
			if(currLineCode=="I" && currTargetUrl.indexOf("portableBuyFlowBegin")!=-1 && "false" == "true")//check portable package
				checkPackage=true;	
			if(currLineCode=="I" && currTargetUrl.indexOf("hispeedDSLBuyFlow")!=-1 && "false" == "true")//check DSL package
				checkPackage=true;	
			 
			 
			 var linkCableBuyflow=false;
			 var hispeedCableAddressEqual = false;
			 if(currLineCode=="I" && currTargetUrl.indexOf("hispeedCableBuyFlow")!=-1){
			 	 linkCableBuyflow=true;
			  	
			  	var inputLastName=getCompareString(document.getElementById("a_txtLastName").value);
			  	var inputStreetNum=getCompareString(document.getElementById("a_txtStreetNum").value);
			  	var inputAptNum=getCompareString(document.getElementById("a_txtAptNum").value);
			 	hispeedCableAddressEqual=(shoppingLastName==''||getCompareString(shoppingLastName)==getCompareString(inputLastName))
			 			&&(shoppingStreetNum==''||shoppingStreetNum==inputStreetNum)
			 			&&(shoppingAptNum==''||shoppingAptNum== inputAptNum)
			 			&&(getCompareString(inputPostalCode)==getCompareString(shoppingPostalCode)
			 		);	
			 	
			  }
			  
			_boxServiceCheck1Close();  

			if(shoppingPostalCode=="null" || shoppingPostalCode=="")
				retu = true;
			else if(inBrowseFlow)
				retu = true;
			else{ 
				if(linkCableBuyflow && hispeedCableAddressEqual && false){
					showProductInCartBox();
					retu = false;
				}else if(linkCableBuyflow && !hispeedCableAddressEqual && false){
					showAddressNotEqualBox();
					retu = false;
				}else if( getCompareString(inputPostalCode)==getCompareString(shoppingPostalCode) && checkPackage==true){	
					showProductInCartBox();
					retu = false;			
				}else if(inputPostalCode!=shoppingPostalCode ){
					showAddressNotEqualBox();
					retu = false;
				}else{
					retu = true;
				}
			}
			
			if(retu){
				var productLine = "";
				if(currLineCode=="H")
					productLine = "Homephonoe";
				else if(currLineCode=="I")
					productLine = "Internet";
				else
					productLine = "Unknown";
				
				//Added by Victor on July 05, 2007 for coremetrics
				//cmCreateManualLinkClickTag('','ServiceabilityCheck:'+productLine);
				
//Coremetrics tag added by Nenad P. on March 10, 2008
  				
			}
			return retu;
		}
		
		return false;

	}

var isNN = (navigator.appName.indexOf("Netscape")!=-1);

function autoTab(input,len, e) {
  var keyCode = (isNN) ? e.which : e.keyCode; 
  var filter = (isNN) ? [0,8,9] : [0,8,9,16,17,18,37,38,39,40,46];
  if(input.value.length >= len && !containsElement(filter,keyCode)) {
    input.value = input.value.slice(0, len);
    input.form[(getIndex(input)+1) % input.form.length].focus();
  }
  function containsElement(arr, ele) {
    var found = false, index = 0;
    while(!found && index < arr.length)
    if(arr[index] == ele)
    found = true;
    else
    index++;
    return found;
  }
function getIndex(input) {
    var index = -1, i = 0, found = false;
    while (i < input.form.length && index == -1)
    if (input.form[i] == input)index = i;
    else i++;
    return index;
  }
  return true;
}
