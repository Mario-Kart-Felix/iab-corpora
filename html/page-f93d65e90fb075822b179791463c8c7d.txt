            return true;

            } else {
                $("#validmodel").css('display', 'none');
            }

        });


        $('input#quotezip').keyup(function(e) {

                $("#validzip").css('display', 'block');
                $("#ziprequired").css('display', 'block');
            } else 
            {
                $("#validzip").css('display', 'none');
                $("#ziprequired").css('display', 'none');
            }
        });


        $("#NewCarList").change(function() {

            var make = $("#NewCarList").val();
            $("#validmodel").css('display', 'block');

            if (make == "") {
                $("#YearModel").attr('disabled', true).val('');
            }
            else {
                $("#validmake").css('display', 'none');
                $.ajax(
                    {
                        type: "GET",
                        url: '/Cars/' + make + '/GetYearModelListByMake',
                        //url: '/Cars/' + make + '/GetYearModelListByMake',
                        dataType: 'json',
                        timeout: 210000,
                        contentType: "application/json; charset=utf-8",
                        error: function(request, error) { $("#YearModel").attr('disabled', true).val(''); },
                        success: function(yearModels) {
                            $("#YearModel").html("");
                            var yearModelIndex = 0;
                            if (yearModels.length > 0) {
                                var options = "<SELECT tabindex='2' style='width:180px;height:17px;'  id='YearModel'><option value=''>Select Model</option>";

                                for (var itemIndex = 0; itemIndex < yearModels.length; itemIndex++) {
                                    var yearModel = yearModels[itemIndex];
                                    options += "<option value='" + yearModel.Value + "'>" + yearModel.Name + "</option>";
                                }
                                options += '</SELECT>';
                                $("#YearModel").removeAttr('disabled').html(options);
                            } else {
                                $("#YearModel").attr('disabled', true).val('');
                            }

                            $("#YearModelDiv").html(options);
                        }
                    });
            }
        });      //Year

    function SetEnterKey(e) {
        
        var key;
        if (window.event)
            key = window.event.keyCode;     //IE
        else
            key = e.which;     //firefox

        if (key == 13) {
            var btn = document.getElementById('goquoteresearchcarimage');
            if (btn != null) {
                btn.click();
                return false;
            }
        }
    };

    function updateMessageCH(msg) {
        var message = document.getElementById("valMessageCH");

    }

    function checkLengthCH(zipcode, n, min) {
        if (zipcode.length < min) {
            updateMessageCH("Length of " + n + " must be " + min + " digit.");
            return false;
        } else {
            return true;
        }

    }

    function checkRegexpCH(zipcode, regexp, n) {
        if (!(regexp.test(zipcode))) {
            updateMessageCH(n);
            return false;
        } else {
            return true;
        }
    }


    function checkZipcodeCH() {

        var bValid = true;
        var zipCode = $("#txtZipcodeCH").val();

        bValid = bValid && checkLengthCH(zipCode, "Zip Code", 5);
        bValid = bValid && checkRegexpCH(zipCode, /^\d{5}$/, "Please enter a valid 5 digit ZIP code.");
        if (bValid) {

            $.post("/Page/VerifySaveZipCode", { zipCode: zipCode }, function(data) {
                if (data == false && $('.changethiszipcode').length > 0) {
                    updateMessageCH("Please enter a valid zip code.");
                }
                else {
                    $.post("/Page/ZipCode",
                    { zipCode: zipCode }, function(data) {

                    NADAjs.AutoTraderWidget.SearchNewZip(zipCode);

                    if ($('.changethiszipcode').length > 0) {
                        $('.changethiszipcode').html(zipCode);
                        NADADealerQuotejs.DealerQuote.GetNewDealers(zipCode);
                    }
                });

                    $.modal.close();

                }
            });
        }
    }
    
    function CheckCookiesCH() {
        document.cookie = "Enabled=true";
        var cookieValid = document.cookie;
        if (cookieValid.indexOf("Enabled=true") != -1) {
            $("#txtZipcodeCH").val(getCookieCH("zipcode"));
        }
    }

    function getCookieCH(c_name) {
        if (document.cookie.length > 0) {
            c_start = document.cookie.indexOf(c_name + "=");
            if (c_start != -1) {
                c_start = c_start + c_name.length + 1;
                c_end = document.cookie.indexOf(";", c_start);
                if (c_end == -1) c_end = document.cookie.length;
                return unescape(document.cookie.substring(c_start, c_end));
            }
        }
        return "";
    }
