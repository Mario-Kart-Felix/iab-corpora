function manageColors (disable){
	for (var i=0; i<20; i++){
		var id = 'color_'+i;
		var obj = document.getElementById (id);
		if (obj != undefined) obj.disabled = !disable;
	}
}

function enablingInputs (name, disable){
	var obj = document.getElementById (name);
	if (obj != undefined){
		obj.disabled = !disable;
	}
	else{
		if (name=='color') return manageColors (disable);
		var obj = document.getElementById (name+'_min');
		if (obj != undefined) obj.disabled = !disable;
		var obj = document.getElementById (name+'_max');
		if (obj != undefined) obj.disabled = !disable;
	}
}

function animateFilter (obj){
	var name=obj.getAttribute('sname');
	var block = document.getElementById('block_'+name);
	if (block.style.display == 'none'){
		block.style.display = 'block';
		enablingInputs (name, true);
		obj.className = 'expand open';
	}
	else{
		block.style.display = 'none';
		enablingInputs(name, false);
		obj.className = 'expand';
	}
}

function showOrHideFilters (){
	var block = document.getElementById('filter_list');
	if (block.style.display == 'none'){
		block.style.display = 'block';
	}
	else{
		block.style.display = 'none';
	}
}
