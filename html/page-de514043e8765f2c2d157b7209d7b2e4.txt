  	function positionSplash() {
		var windowDimensions = document.viewport.getDimensions();
		var windowHeight = windowDimensions.height;
		var windowWidth = windowDimensions.width;
		
		var splashDimensions =  $('sti_splash').getDimensions();
		
		var splashPositionTop = windowDimensions.height/2 - splashDimensions.height/2;
		if (splashPositionTop < 0)
			splashPositionTop = 0;
		var splashPositionLeft = windowDimensions.width/2 - splashDimensions.width/2;
		if (splashPositionLeft < 0)
			splashPositionLeft = 0;
			
		$('sti_splash').setStyle({left: splashPositionLeft + 'px', top: splashPositionTop + 'px'});
	}
	
	function shouldShowSplash() {
		var hasSubscribedToDeal = Cookie.read("date_subscribed_to_deal_email");
		var hasClosedDealSplash = Cookie.read("date_closed_splash");
		return !(hasSubscribedToDeal || hasClosedDealSplash);
	}
	
	function showSplash() {
		if (shouldShowSplash()) {
			positionSplash();
			$('sti_overlay').style.display = 'block';
			$('sti_splash').style.display = 'block';
			window.onresize = positionSplash;
		}
	}
	
	function hideSplash() {
		var currentDate = new Date();
		var expiryDate = new Date( currentDate.getTime() + 63113851900 );
		Cookie.write({name : "date_closed_splash", expires : expiryDate, path : '/', value : currentDate.getTime()});
		$('sti_splash').style.display = 'none';
		$('sti_overlay').style.display = 'none';
	}

  <script language = "javascript">
    var DEFAULT_ENTER_EMAIL_TEXT = "Enter Email Address";
    var DEFAULT_ENTER_CITY_TEXT = "Enter City";   
    var INVALID_EMAIL_ERROR_MESSAGE = "Please enter a valid email address.";
    var INVALID_POSTAL_ERROR_MESSAGE = "Please enter a valid postal code.";
    var BLANK_POSTAL_ERROR_MESSAGE = "Please enter a postal code.";
    var BLANK_EMAIL_ERROR_MESSAGE = "Please enter an email address.";
    var INVALID_CITY_ERROR_MESSAGE = "Please enter a valid city.";
    var BLANK_CITY_ERROR_MESSAGE = "Please enter a city.";
    var DEFAULT_GENERAL_ERROR_MESSAGE = "There was an error completing your request. Please try again.";

    function isValidEmail() {
    	var email = '';
    	if ($('splashEmailInput') != null)
    		email = $('splashEmailInput').value;
    	return /^[-\w.]+@([-\w]+[.])+[A-Za-z]{2,4}$/.test(email);
    }

    function isValidPostal() {
    	var postalCode = '';
    	if ($('splashPostalInput') != null)
    		postalCode = $('splashPostalInput').value;
    	return /^[A-z][0-9][A-z]\s*[0-9][A-z][0-9]$/.test(postalCode);
    }

    function isValidCity() {
    	var city = '';
    	if ($('splashCityInput') != null)
    		city = $('splashCityInput').value;
    	if (/^Enter City$/.test(city))
    		city = '';
    	return /^[-A-Za-z' ,]+$/.test(city);
    }

    function validateEmail() {
    	if (isValidEmail()) {
    		$('emailError').style.display = 'none';
    	} else {
    		var email = '';
    		if ($('splashEmailInput') != null)
    			email = $('splashEmailInput').value;
    		
    		var defaultEmailTextRegex = new RegExp("^" + DEFAULT_ENTER_EMAIL_TEXT + "$");
    		if (defaultEmailTextRegex.test(email))
    			email = '';
    			
    		if (email != null && email.length > 0)
    			$('emailError').innerHTML = INVALID_EMAIL_ERROR_MESSAGE;
    		else
    			$('emailError').innerHTML = BLANK_EMAIL_ERROR_MESSAGE;
    			
    		$('emailError').style.display = 'block';
    	}
    }

    function validatePostal() {
    	if (isValidPostal()) {
    		$('postalError').style.display = 'none';
    	} else {
    		$('postalError').innerHTML = INVALID_POSTAL_ERROR_MESSAGE;
    		$('postalError').style.display = 'block';
    	}
    }

    function validateCity() {
    	if (isValidCity()) {
    		resetCityError();
    	} else {
    		var city = '';
    		if ($('splashCityInput') != null)
    			city = $('splashCityInput').value;
    		
    		var defaultCityTextRegex = new RegExp("^" + DEFAULT_ENTER_CITY_TEXT + "$");
    		
    		if (defaultCityTextRegex.test(city))
    			city = '';
    		
    		if (city != null && city.length > 0)
    			$('cityError').innerHTML = INVALID_CITY_ERROR_MESSAGE;
    		else
    			$('cityError').innerHTML = BLANK_CITY_ERROR_MESSAGE;
    		
    		$('cityError').style.display = 'block';
    	}
    }

    function redirectUser() {
    	$('closeButton').onclick=function() {};
    	$('emailContainer').style.display = 'none';
    	$('thankyouDiv').style.display = 'block';
    	$('splashCloseContainer').style.display = 'none';
    	$('splashCloseContainerPlain').style.display = 'block';
    }

    function showThankYou() {
    $('generalError').style.display = 'none';
    	$('splashTitleDefault').style.display = 'none';
    $('splashTitleThankYou').style.display = 'block';
    	$('emailContainer').style.display = 'none';
    	$('thankyouDiv').style.display = 'block';
    	$('splashCloseContainer').style.display = 'none';
    	$('splashCloseContainerPlain').style.display = 'block';
    	$('Autocomplete_splashCityInput').style.display = 'none';
    }

    function showError(generalError, cityError, emailError) {
    	if (generalError != null) {
    		$('generalError').innerHTML = generalError;
    		$('generalError').style.display = 'block';
    	}
    	
    	if (emailError != null) {
    		$('emailError').innerHTML = emailError;
    		$('emailError').style.display = 'block';
    	}
    	
    	if (cityError != null) {
    		$('cityError').innerHTML = cityError;
    		$('cityError').style.display = 'block';
    	}
    }

    function resetEmailError() {
    		$('emailError').innerHTML = '&nbsp';
    		$('emailError').style.display = 'none';
    }

    function resetCityError() {
    		$('cityError').innerHTML = '&nbsp';
    		$('cityError').style.display = 'none';
    }

    function resetGeneralError() {
    		$('generalError').innerHTML = '&nbsp';
    		$('generalError').style.display = 'none';
    }

    var haveSubmittedForm = false;

    function showDealSignup() {
    haveSubmittedForm = false;
    	$('splashTitleDefault').style.display = 'block';
    	$('emailContainer').style.display = 'block';
    	$('splashCloseContainer').style.display = 'block';
    	$('splashCloseContainerPlain').style.display = 'none';
    	$('spinnerDiv').style.display = 'none';
    $('dealSignupForm').style.display = 'block';
    }

    function showSpinner() {
    $('generalError').style.display = 'none';
    $('dealSignupForm').style.display = 'none';
    	$('spinnerDiv').style.display = 'block';
    	$('splashCloseContainer').style.display = 'none';
    	$('splashCloseContainerPlain').style.display = 'none';
    }

    function submitSignup() {
    	if (!haveSubmittedForm) {
      	haveSubmittedForm = true;
    	
    		showSpinner();
    	
    		//var url = '/dealSignup';
    		var url = '/shop/dealSignup?dealSignupUrl=' + encodeURIComponent(document.URL).replace(/_/g, '%5F').replace(/-/g, '%2D') + '&email=' + encodeURIComponent($('splashEmailInput').value).replace(/_/g, '%5F').replace(/-/g, '%2D') + '&city=' + encodeURIComponent($('splashCityInput').value).replace(/-/g, '%2D');
    	new Ajax.Request(url, {
    			method: 'get',
    			onComplete: function(response) {
    				var responseObject = response.responseJSON;
    				var responseStatus = response.status;
      			if (responseStatus == 200 && responseObject != null && responseObject.was_successful) {
     					successful = true;
     				Cookie.write({name : "date_subscribed_to_deal_email", expires : '', path : '/', value : new Date()});
      				closeCounter = 7;
      				showThankYou();
      				countDownClose();
      			} else {
      				closeCounter = 7;
      				if (responseObject != null)
      					showError(responseObject.general_error, responseObject.city_error, responseObject.email_error);
      				else
      					showError(DEFAULT_GENERAL_ERROR_MESSAGE);
      				showDealSignup();
      			}
    			}
    	});
    }
    }

    var closeCounter = 7;

    function submitDealSignup() {
    	validateEmail();
    	validateCity();
    	if (isValidEmail() && isValidCity()) {
    		submitSignup();
    	}
    	return false;
    }

    function countDownClose() {
    	if (closeCounter == 0) {
    		hideSplash();
    	} else {
    		$('countdownClose').innerHTML = closeCounter;
    		--closeCounter;
    		var t=setTimeout("countDownClose()",1000);
    	}
    }

    function submitenter(myfield,e) {
    	var keycode;
    	if (window.event)
    		keycode = window.event.keyCode;
    	else if (e) 
    		keycode = e.which;
    	else 
    		return true;

    	if (keycode == 13) {
    		return submitDealSignup();
    	} else {
    		return true;
    	}
    }

    function countDownShowSignup() {
    	if (closeCounter == 0) {
    		showDealSignup();
    	} else {
    		$('errorCountdownClose').innerHTML = closeCounter;
    		--closeCounter;
    		var t=setTimeout("countDownShowSignup()",1000);
    	}
    }
