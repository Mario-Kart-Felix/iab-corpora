          function ProcessListText(text)
          {			
            var charLimit = 200;
            var realText = removeHTMLTags(text);
	       
		    text = htmlEntities(text);
	        if(realText.length < charLimit)

	        var words = realText.substring(0, charLimit);
			var isCompleteWord = false;
			
			if(/\/|\.|\,|\?!|\s/.test(realText.charAt(charLimit-1))) isCompleteWord = true;
			
			for(var j = charLimit; j < realText.length; j++)
			{
				if(isCompleteWord) break;
				if(!isCompleteWord && /\S/.test(realText.charAt(j)) && /[^\/\.\,\?!\s]/.test(realText.charAt(j)))

					words = words + realText.charAt(j);
				}else
				{
					isCompleteWord = true;
				}
			}
			
			if(/\s/.test(words.charAt(words.length - 1)))
			{
				words = words.substring(0, words.length - 1);
			}
			
	        var returnString = '';
	        var counterWalk = 0;
	        var tagClose = true;			
			var tagOpen = false;
			var moreLink = '...';
			
	        for(var i=0; i < text.length; i++)

	            }
	            else
	            {
	              if(text.charAt(i) == '>')
	              {
	                tagClose = true;
					tagOpen = false;

				  else if (text.charAt(i) == '<')

		  function htmlEntities(strInputCode)
		  {
			strInputCode = strInputCode.replace(/&(lt|gt);/g, function (strMatch, p1){
 		 	      return (p1 == "lt")? "<" : ">";
 		          });
				  
			return strInputCode
					.replace(/(^\s*)|(\s*$)/gi,"")
					.replace(/&nbsp;/g, " ")
					.replace(/&amp;/g, "&")
					.replace(/\s+/g," ")
					.replace(/(\r\n|\n|\r)/gm, "");
		  }
		  
          function removeHTMLTags(strInputCode)
          {
	          strInputCode = strInputCode.replace(/&(lt|gt);/g, function (strMatch, p1){
 		 	      return (p1 == "lt")? "<" : ">";
 		          });
 		      return strInputCode
						.replace(/<\/*[a-z][^>]+?>/gi, "")
						.replace(/<\/[^>]+(>|$)/g, "")
						.replace(/&nbsp;/g, " ")
						.replace(/&amp;/g, "&")
						.replace(/<b>/gi, "")
						.replace(/(^\s*)|(\s*$)/gi,"")
						.replace(/\s+/g," ")
						.replace(/(\r\n|\n|\r)/gm, "");			  
          }
