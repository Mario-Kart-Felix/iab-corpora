     <script>
     function createCookie(name,value,days) 
     {
            if (days) {
                var date = new Date();
                date.setTime(date.getTime()+(days*24*60*60*1000));
                var expires = "; expires="+date.toGMTString();
            }
            else var expires = "";
                document.cookie = name+"="+value+expires+"; path=/";
      }


      function getCookie(cookieName)
      {
        var cookieValue = document.cookie;
            var cookieStartsAt = cookieValue.indexOf(" " + cookieName + "=");
            if (cookieStartsAt == -1)
            {
            cookieStartsAt = cookieValue.indexOf(cookieName + "=");
            }
            if (cookieStartsAt == -1)
            {
            cookieValue = null;
            }
            else
            {
            cookieStartsAt = cookieValue.indexOf("=", cookieStartsAt) + 1;
            var cookieEndsAt = cookieValue.indexOf(";", cookieStartsAt);
            if (cookieEndsAt == -1)
            {
                cookieEndsAt = cookieValue.length;
            }
            cookieValue = unescape(cookieValue.substring(cookieStartsAt,
            cookieEndsAt));
            }
            return cookieValue;
     }

        $("#quick_search").css('text-align','left');
    }
}

    $("#quick_search").val('');
    $("#quick_search").css('text-align','left');

    if( $("#quick_search").val() == globalSearchBoxDefault){
        $("#quick_search").val('');

    }

$().ready(function() {

    $('#postal_code_1').focus();
	
    try{
	    $("#quick_search").autocomplete("/search/dim_results.jsp", {
	        max: 30,
	        minChars: 3,

	        width: 250,
	        multiple: false,
	        scroll: true,
	        scrollHeight: 300,
	        formatItem: formatItem,
	        formatResult: formatResult,
	        selectFirst: false
	    });
    }catch(err){ 	}
    
    
    try{

    });
    }catch(err){ 	}
});

function showFlyout(flyoutObj){
    $(flyoutObj).find('a:first').addClass('active');
    
    if( $(flyoutObj).find('a:first').outerWidth(true) > $(flyoutObj).find('div.flyout').outerWidth(true)){
        $(flyoutObj).find('div.flyout').width($(flyoutObj).find('a:first').outerWidth(true) - 22);
    }
    
    $(flyoutObj).find('div.flyout').show();
    $('#quick_search').blur();
    $('#postal_code_1').focus();
}

function nav(){
    var mainNavWidth = 993;
    var subNavWidth = 820;
    var mainNav =  $('#dept_nav');
    var showTimer = null;
    
    $('#dept_nav > li').mouseover(function() {
        
        var navItem = $(this);
        
        if (showTimer){
            clearTimeout(showTimer);
        }
        showTimer = setTimeout(function(){
            subNavWidth = $(navItem).find('ul:first').outerWidth(true);
            
            if( ($(navItem).offset().left - $(mainNav).offset().left) > (mainNavWidth - subNavWidth) ){
            	$(navItem).find('ul:first').css('left', ($(mainNav).offset().left + mainNavWidth - subNavWidth) + 'px');
            } else {
                $(navItem).find('ul:first').css('left', $(navItem).offset().left + 'px');
            }
            
            $(navItem).addClass('activeOver');
            
            $(navItem).find('ul:first').css('visibility', 'visible');
               
            formatHowToMenu();
            
        }, 500);
    });
    
	
    
    
    
    $('#dept_nav > li').mouseleave(function() {
        
        clearTimeout(showTimer);
        
        if( $(this).hasClass('categoryExt') ){
        	$('#dept_nav li ul.more_category').css('visibility', 'hidden');
        } else {
        	$('#dept_nav li ul.nav_sub_category').css('visibility', 'hidden');
        }
        
        $(this).removeClass('activeOver');
        
    });
    
    /////////////////////////// MY STORE POPUP //////////////////////////////////////
    $('li.flyout').mouseover(function() {
        showFlyout($(this));
    });
    
    $('li#myStoreSelected.flyout').mouseleave(function() {
        $(this).find('a:first').removeClass('active');
        $(this).find('div.flyout').hide();
        $('#postal_code_1').blur();
        $('#quick_search').focus();
    });

    $('.closeBtn').live("click", function(){
    	var findStoreElem = $('li.flyout');
    	$(findStoreElem).find('a:first').removeClass('active');
        $(findStoreElem).find('div.flyout').hide();
        $('#postal_code_1').blur();
        $('#quick_search').focus();
    });
    /////////////////////////// MY STORE POPUP END //////////////////////////////////
}

/* Adds formatting to the HOW-TOs banner in top nav dropdowns */
function formatHowToMenu(){		
		
	$(".nav_sub_category_howtos_banner").each(function(){						
		
		var subcat_width = $(this).parent().width();		
		var logo_container_width = $(this).find(".howtos_banner_logo").width();
		
		var links_container_width = subcat_width - logo_container_width - 30;
				
		$(this).width(subcat_width);
				
		$(this).find(".howtos_banner_links").width(links_container_width);
		$(this).find(".howtos_banner_links ul").width(links_container_width);
		
		if($(this).find(".howtos_banner_links ul").height() < 35){
			$(this).find(".howtos_banner_links ul li").not('[class="howtos_banner_title"]').css("margin-top", "10px");
		}
	})
}
