	window.onfocus = function() { updateSessionTimeout(true); }
	
	/* need to create alias because of jkmegamenu */
	var $jq = jQuery.noConflict();
	
	$jq(function($jq) {       
		$jq('<div id="sessionTimeoutDiv" title="Session Timeout"></div>').appendTo('body');
		
		$jq( "#sessionTimeoutDiv" ).dialog({
	        autoOpen: false,
	        show: "fade",
	        hide: "fade",
	        position: {my: "center", at: "top+30%"},
	        modal: true,
	        close: function() {
	        	sessionTimeoutContinue();
	        }
		});
		
		updateSessionTimeout(false);
		
	});
	
	
 	function showTimeoutMsg() {
		
		var msg = 'Your session will expire in 3 minutes. ' +
			'To continue working, click on the "OK" button below. <br /><br />' +
    		'Note:<br />' + 
   			'This is being done to protect your privacy. Unsaved changes will be lost.'
   			
		msg += '<br /><br />' +
			'<div align="center">' +
			'<input type="button" value="  OK  " onclick="sessionTimeoutContinue();" />' +
			'</div>';
		
		document.getElementById( "sessionTimeoutDiv" ).innerHTML = msg;
		$jq( "#sessionTimeoutDiv" ).dialog( "open" );
		return false;
	}
	
	
	function updateSessionTimeout( check ) {
		var url = top.location.href;
		var isSearch = ( ( url.match( /search-grants.html$/ ) != null ) || ( url.match( /view-opportunity.html$/ ) != null ) );
		
		var id = getGGUserProfileUID();
		if ( id == "" || isSearch ) { return; }
		var queryStr = 'jp={"id":"' + id + '"';
		if ( check == false ) { queryStr += ',"check":false'; }
		queryStr += '}';
		
		$jq.ajax({
			"url": "http://www.grants.gov/grantsws/SessionTimeout",
			"data": queryStr,
			"dataType": "jsonp",
			"error": function( error ) {},
			"success": function( data ) {
				var id = data.id;
				var timeRemainingInMillis = data.timeRemainingInMillis;
				
				if ( timeRemainingInMillis == 0 ) {
					sessionTimeoutLogin();
				} else {
					updateSessionTimer( timeRemainingInMillis );
				}
			}
		});	
	}
	
	
	var sessionTimeoutTimerObj;
	
	function updateSessionTimer( timeInMillis ) {
		clearTimeout( sessionTimeoutTimerObj );
		var sessionTimeoutWarningInMillis = 3 * 60 * 1000;
		if ( timeInMillis > sessionTimeoutWarningInMillis ) {
			var timeToWarningInMillis = timeInMillis - sessionTimeoutWarningInMillis;
			/* set difference to display warning */
			sessionTimeoutTimerObj = setTimeout( "updateSessionTimeout(true)", timeToWarningInMillis );
			$jq( "#sessionTimeoutDiv" ).dialog( "close" );
		} else {
			sessionTimeoutTimerObj = setTimeout( "updateSessionTimeout(true)", timeInMillis );
			showTimeoutMsg();
		}// if-else
		
	}
	
	
	function sessionTimeoutContinue() {
		updateSessionTimeout( false );
		$( "#sessionTimeoutDiv" ).dialog( "close" );
	}
	
	
	function sessionTimeoutLogin() {
		top.location.href = "/custom/spoeeLogout.jsp";
	}

	function populateFacetTab( jsonFacetOptions, facetName, divTarget ) {
		var FUNCTION_NAME = "[populateFacetTab()] ";
		var divObj = document.getElementById( divTarget );
		divObj.innerHTML = "";
		
		//alert (FUNCTION_NAME + facetName + divTarget);
		var count = jsonFacetOptions.length;
		//alert( FUNCTION_NAME + facetName + " count=" + count );

		var displayStr = '<table width="100%" style="font-size:12px;" cellspacing="4" cellpadding="4" border="0"><tbody><tr><td valign="top">';
		var option = '';
		
		if ( count == 0 ) {
			displayStr += "No facet values found.";
			
		} else if (facetName != ELIGIBILITY_FACET_NAME ) {
			//alert("FACETNAME= " + facetName);
			
			var half = Math.round( count/2 );
			//alert( FUNCTION_NAME + count + " --- half: " + half );
			
			for ( var i = 0; i < count; i++ ) {
				option = jsonFacetOptions[i];
				displayStr += createLink( facetName, option );
					
				if ( i == ( half - 1 ) ) {
					displayStr += '</td><td valign="top">';
				}// if
				
			}// for

		} else {
			
			for ( var i = 0; i < count; i++ ) {
				option = jsonFacetOptions[i];
				displayStr += createLink( facetName, option );
				
			}// for
		
		}// if-else
		
		displayStr += '</td></tr></tbody></table>';
		
		//alert( FUNCTION_NAME + ": " + displayStr );
		divObj.innerHTML = displayStr;
		
	}// populateFacetTab

		} catch ( error ) {
			alert( "error creating link: " + error.description );
		}// try-catch
		
	}// createLink
