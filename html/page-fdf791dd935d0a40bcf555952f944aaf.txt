addComma = function (x) {
x = x.toString();
return [x.slice(0, x.length - 3), x.slice(x.length - 3)].join(',');
},
isIn = function (x, y) {
return y.indexOf(x) != -1;

scrub = function(key, v){
if (key && v) {
switch (key) {
case 'price':
if (parseInt(v.replace('$',''), 10) > 0) { // checks to make sure price isn't 0, .replace makes sure a number is returned if price begins with '$'
v = v.split('.')[0];
v = ((isIn('$', v)) ? '' : '$') + ((isIn(',', v)) ? v : addComma(v));  // TODO: add internationalized currency
} else {
v = "";
}
default: break;
}
return v;
}

// builds links for a vehicle
buildLink = function(vehicle, urlType) {
var url = urls[urlType],
link = {
url:'',
title:[]
}, param, params;
if (url != urls['404']) {
switch (urlType) {
case "Request A Quote":
case "Showroom":
if (url.indexOf('/') == 0) {
url = url.slice(1);
}
url = url.replace('/index.htm', '').replace('/results.htm','').split('/');
url.push(vehicle.year, vehicle.make, vehicle.model, vehicle.bodyStyle);
link.url = '/' + url.join('/') + '.htm';
link.title.push(vehicle.year);
break;
case "New Inventory":
params = {
make: vehicle.make,
model: vehicle.model,
bodyStyle: vehicle.bodyStyle,
queryString: []
};
for (param in params) {
if (param != "queryString") {
params.queryString.push(param + '=' + params[param]);
}
}
link.url = url + (url.indexOf('?') == -1 ? '?' : '&') + params.queryString.join('&');
break;
case "Certified Pre-Owned":
//JIRA-15041 - purposely not breaking here, so that this condition is for certified used and for used
case "Pre-Owned":
if (vehicle.model.indexOf("1") == 0) {
vehicle.model = "1+Series"; 
}
else if (vehicle.model.indexOf("3") == 0) {
vehicle.model = "3+Series"; 
}
else if (vehicle.model.indexOf("5") == 0) {
vehicle.model = "5+Series"; 
}
else if (vehicle.model.indexOf("6") == 0) {
vehicle.model = "6+Series"; 
}
else if (vehicle.model.indexOf("7") == 0) {
vehicle.model = "7+Series"; 
}
else if (vehicle.model.indexOf("X") == 0) {
vehicle.model = "X+Series"; 
}
else if (vehicle.model.indexOf("M") == 0) {
vehicle.model = "M+Series"; 
}
params = {
make: vehicle.make,
model: vehicle.model,
bodyStyle: vehicle.bodyStyle,
queryString: []
};
for (param in params) {
if (param != "queryString") {
params.queryString.push(param + '=' + params[param]);
}
}
link.url = url + (url.indexOf('?') == -1 ? '?' : '&') + params.queryString.join('&');
break;
case '404':
break;
default: break;
}
if (!!link.url) {
link.title.push(vehicle.make, vehicle.model, vehicle.bodyStyle, (urlType != "Request A Quote" ? urlType : ''));
link.title = link.title.join(' ');
}
}
return (!!link.url) ? link : '';
},
// fires after .buckets li a is clicked and ajax call returns
buildBucketHtml = function(bucket) {
var html = '';
html += '<h4>The '+ buckets[bucket][0].make + ' ' + bucket + ((bucket.length == 1) ? ' Series' : '') + '</h4>';
html += '<ul>',
setBodstyles = function(bucket){
for (var i = 0; i < buckets[bucket].length; i++) {
var bodystyle = buckets[bucket][i].bodyStyle;
if( jQuery.inArray( bodystyle, bodystyles ) < 0 ){
bodystyles.push(bodystyle)
}
}
return bodystyles;
},
bodystyles = [];
bodystyles = setBodstyles(bucket);
for(var i = 0; i < bodystyles.length; i++){
var current = true;
html += '<li' ;
html += ( i === 0 ) ? ' class="active">' :'>';
html += '<ul>';
html += '<li class="root">';
html += bodystyles[i]
html += '</li>';
for (var x = 0; x < buckets[bucket].length; x++) {
var vehicle = buckets[bucket][x];
if(vehicle.bodyStyle === bodystyles[i]){
if( i === 0 && current){
html += '<li class="current"><a href="#'+ x +'">' + vehicle.model + ' ' + vehicle.bodyStyle +'</a></li>';
current = false;
}else{
html += '<li><a href="#'+ x +'">' + vehicle.model + ' ' + vehicle.bodyStyle +'</a></li>';
}
}
}
html +='</ul></li>';
}
html += '</ul>';
return html;
},
// fires after .modelBucket a is clicked
buildModelHtml = function (vehicle) {
var html = '',
urlType, link;
html += '<div class="details"><h5> Shopping Tools </h5>';
html += '<ul class="specs">';
if (vehicle.price["msrp"]) {
html += '<li>Starting at: ' + scrub('price', vehicle.price.msrp) + '</li>';
}
html += '<li>' + vehicle.hp + 'hp</li>';
html += '<li>' + vehicle.mpg.min  + '-' + vehicle.mpg.max + 'mpg</li>';
html += '</ul><ul class="links">';
for (urlType in urls) {
if (urlType != "ajaxTarget") {
link = buildLink(vehicle, urlType);
if (!!link.url) {
html += '<li><a href="' + link.url + '"' + (!!link.title ? (' title="' + link.title + '"') : '') + '>' + urlType + '</a></li>';
}
}
}
html += '</ul></div>';
html += '<div class="media">';
if (!preloadImages) {
html += '<img src="' + vehicle.images.full + '"/>';
}
html += '</div>';
return html;
