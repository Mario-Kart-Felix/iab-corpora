        function doVehicleSearch() {
            var searchType = "";
            var radioGroup = document.getElementsByName("inventoryType");
            for( i = 0; i < radioGroup.length; i++ ) {
                if( radioGroup[i].checked == true ) {
                    searchType = radioGroup[i].value;
                }
            }
            var yearOptions = document.getElementById("yearOptions");
            var makeOptions = document.getElementById("makeOptions");
            var priceOptions = document.getElementById("priceOptions");
            var options = "";
            if ( yearOptions[yearOptions.selectedIndex].value != "-1" ) {
                options += "year=" + yearOptions[yearOptions.selectedIndex].value + "&";
            }
            if ( makeOptions[makeOptions.selectedIndex].value != "-1" ) {
                options += "ddlMake=" + makeOptions[makeOptions.selectedIndex].value + "&";
            }
            if ( priceOptions[priceOptions.selectedIndex].value != "-1" ) {
                options += "maxPrice=" + priceOptions[priceOptions.selectedIndex].value + "&";
            }
            vehicleSearch(searchType, options);
        }
        function getYouTubeThumbnail( url, size ) {
            if (url === null) {
                return "";
            }
            size = (size === null) ? "big" : size;
            var vid;
            var results;
            results = url.match("[\\?&]v=([^&#]*)");
            vid = ( results === null ) ? url : results[1];
            if (size == "small") {
                return "http://img.youtube.com/vi/"+vid+"/2.jpg";
            } else {
                return "http://img.youtube.com/vi/"+vid+"/0.jpg";
            }
        }
