        bTrackToyotaLeadType += ', ';
    }
    bTrackToyotaLeadType += type;
}
  
function getPageName(){
    var name = $('input[name="__pageName"]').first().val();
	
    if(String.isNullOrWhiteSpace(name))
    {
        name = window.locations.pathname.toLowerCase();
        name = name.replace('/', '');
    }
	
    return name;
}
  
function setLeadInfo(){
    var prop6Data;

    if($('#InvContainer.InvDetails').length > 0)
    {
        isInventoryPage = true;
        prop6Data = 'vehicle details page -      - ';
        //Prop6 should only contain data for one vehicle.
        if (!isSingleVehicle(prop6Data)) {
            prop6Data = 'no value';
        }

        bTrackToyotaLeadName = prop6Data;
        bTrackToyotaLeadType = "inventory";
        tmsomni.events = "event81";
        bTrackToyotaEventStart = 'event54';
        bTrackToyotaEventComplete = 'event55';
    }
    else if($('#InvContainer.InvListings').length > 0)
    {
        isInventoryPage = true;
        prop6Data = 'vehicle listings page -      - ';
        //Prop6 should only contain data for one vehicle.
        if (!isSingleVehicle(prop6Data)) {
            prop6Data = 'no value';
        }

        //TODO: Check search data
        if(dynamicDataHasValue("")){
            tmsomni.prop6 = prop6Data;
            tmsomni.prop64 = "inventory";
            tmsomni.events = "event64"; // only when searching is done
        }
    }
    else if($("form").length > 0)
    {
        //Parse Page name for lead name and type
        var path = getPageName(),
            pathLowered = path.toLowerCase();
		
        if(pathLowered.contains('service')){
            setLeadNameAndType(path, 'service');
            bTrackToyotaEventStart = 'event48';
            bTrackToyotaEventComplete = 'event49';
        }
        else if(pathLowered.contains('finance') || pathLowered.contains('credit')){
            setLeadNameAndType(path, 'finance');
            bTrackToyotaEventStart = 'event52';
            bTrackToyotaEventComplete = 'event53';
        }
        else if(pathLowered.contains('parts')){
            setLeadNameAndType(path, 'path');
            bTrackToyotaEventStart = 'event50';
            bTrackToyotaEventComplete = 'event51';
        }
        else{
            setLeadNameAndType(path, 'other');
            bTrackToyotaEventStart = 'event67';
            bTrackToyotaEventComplete = 'event68';
        }
		
        tmsomni.prop6 = bTrackToyotaLeadName;
        tmsomni.prop64 = bTrackToyotaLeadType;
    }
}
