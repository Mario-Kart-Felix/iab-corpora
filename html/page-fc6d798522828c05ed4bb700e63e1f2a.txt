function request() {
	var http_request = false;
	if (window.XMLHttpRequest) {
		try {
			http_request = new XMLHttpRequest();
		} catch (e) {
			 http_request = false;
		}
		if (http_request.overrideMimeType) {
			http_request.overrideMimeType('text/xml');
		}
	} else if (window.ActiveXObject) {
		// For Internet Explorer on Windows
		try {
			http_request = new ActiveXObject("Msxml2.XMLHTTP");
		} catch (e) {
			 try {
				http_request = new ActiveXObject("Microsoft.XMLHTTP");
			} catch (e) {
				http_request = false;
			}
		}
		if (http_request.overrideMimeType) {
			http_request.overrideMimeType('text/javascript');
		}
	}
	if (!http_request) {
		alert('Cannot create XMLHTTP instance');
		return false;
	} else {
		return http_request;
	}
}

function verifyAdv(f,mode) {
	if (mode == 'detail') {
		var newPage = parseInt(f.seqNum.value);
	} else {
		var newPage = parseInt(f.page.value);
	}
	var finalPage = parseInt(f.lastPage.value);
	if ((newPage > finalPage) || (newPage < 1) || (isNaN(newPage))) {
		if (mode == 'detail') {
   			f.seqNum.value = f.thisPage.value;
		} else {
   			f.page.value = f.thisPage.value;
		}
		f.thisPage.value = "";
		f.lastPage.value = ""; 
	}
}

function validateForm(){
	setAdType(null);
	$.cookie('modifySearch', null);
	return true;

function setAdType(cb) {
	var adType = '';
	if ($('#newCb:checked').length > 0) {
		adType = 'dealer new';
	}
	if ($('#usedCb:checked').length > 0) {
		if (adType) adType += ',';
		adType += "dealer used,liner,aoe_liner,bundle,dealer certified";
	}
	if (!adType) adType = 'dealer new,dealer used,liner,aoe_liner,bundle,dealer certified';
	$('#adTypeID').val(adType);
}

function get_models(arg) {
	var url = "/Autos?category=code%20snippets&tp=TransitCore2007&temp_type=include&tl=3&make=";
	url += arg;

	$('#modelID').empty();
	var item = $('#modelID').get(0);

	if (arg != '') {
		$.getJSON(url, function(response){
			var model_list = response.model_list;
			var selectedIndex = '1';
			if (model_list.length) {
				for (j=0; j < model_list.length; j++) {
					$('#modelID').append(
						$('<option></option>').val(model_list[j].mmcsid).html(model_list[j].model)
					);
				}
				$('#modelID').prepend(
					$('<option></option>').val('0').html('- All Models -')
				);
				$('#modelID').prepend(
					$('<option></option>').val('').html('Select Model').attr('disabled','disabled').css({color: 'gray'})
				);
				item.options[selectedIndex].selected = 'selected';
 				$('#modelID').removeAttr('disabled');
			}
		});
	} else {
		$('#modelID').append(
			$('<option></option>').val('').html('Select Model').attr('disabled','disabled').css({color: 'gray'})
		);
	}
}

		function mycarousel_initCallback(carousel) { // Pause autoscrolling if the user moves with the cursor over the clip.
			carousel.clip.hover(function() {
				carousel.stopAuto();
				}, function() {
				carousel.startAuto();
			});			
				
			$('#mycarousel-next').bind('click', function() {
				carousel.next();
				return false;
			});

			$('#mycarousel-prev').bind('click', function() {
				carousel.prev();
				return false;
			});
				
			$('#mycarousel-next').mouseover(function() {
				$(this).addClass('rightarrow_hover');
				carousel.stopAuto();
			});
			
			$('#mycarousel-next').mouseout(function() {
				$(this).removeClass('rightarrow_hover');
				carousel.startAuto();
			});
	
			$('#mycarousel-next').mouseup(function() {
				$(this).removeClass('rightarrow_click');
				}).mousedown(function() {
				$(this).addClass('rightarrow_click');
			});
	
			$('#mycarousel-prev').mouseover(function() {
				$(this).addClass('leftarrow_hover');
				carousel.stopAuto();
			});
				
			$('#mycarousel-prev').mouseout(function() {
				$(this).removeClass('leftarrow_hover');
				carousel.startAuto();
			});
	
			$('#mycarousel-prev').mousedown(function() {
				$(this).addClass('leftarrow_click');
				}).mouseup(function() {
				$(this).removeClass('leftarrow_click');
			});						
				
		};

function keepHonest(){
	var zipcode = $('#user_zip_code').val();
	if (zipcode.search(/^\d{5}$/)) {
		alert('Please enter a valid 5 digit zip code.');
		return;
	} else {
		document.configStart.submit();
	}
}	
function setUserZipcodeCookie(){
	if ($('#user_zip_code').val() != '') {
		$.cookie('user_zipcode', null);
		$.cookie('user_zipcode',$('#user_zip_code').val());
	}
}

function getUserZipcode(){
	var defaultZip = '';
		defaultZip = '48901';
	var uzc = $.cookie('user_zipcode');
	if (uzc != null && uzc != ''){
		$('#user_zip_code').val(uzc);
		$('#user_zipcode').val(uzc);
		$('#user_zipcode_mpg').val(uzc);
	} else {
		var zipcodeValue = defaultZip;
		var GTCvalue = $.cookie('GTC');
		if (GTCvalue != null){
			var initialChar = GTCvalue.substr(0,1);
			if (initialChar == ':') {
				var reliability = parseInt(GTCvalue.split(':')[1]);
				if (reliability >= 4) {
					var GTCstate = GTCvalue.split(':')[4];
					if (GTCstate == 'MI') {
						zipcodeValue = GTCvalue.split(':')[2];
					}	
				}
			} else {
				var reliability = parseInt(GTCvalue.split(':')[0]);
				if (reliability >= 4) {
					var GTCstate = GTCvalue.split(':')[3];
					if (GTCstate == 'MI') {
						zipcodeValue = GTCvalue.split(':')[1];
					}	
				}
			}
		}
		$('#user_zip_code').val(zipcodeValue);
		$('#user_zipcode').val(zipcodeValue);
		$('#user_zipcode_mpg').val(zipcodeValue);
	}
}

	getUserZipcode();		
}
