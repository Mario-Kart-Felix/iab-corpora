	<script type="text/javascript">
		var speed = 500;
		var pause = 1;
		var timeout = 6000;
		$(".home-cycle").cycle({
			fx: "fadeout",
			speed: speed,
			timeout: timeout,
			slideResize: false,
			delay: speed,
			before: function(){
				setTimeout(function(){
					$("#home-progress").animate({
						width: "100%"
					},
					{
						duration: timeout,
						easing: "linear",
						complete: function(){ $("#home-progress").width(0); }
					});
					pause = speed;
				}, pause );
			}
		});
		var cycleresize = function(){
			$(".home-cycle").show();
			$(".home-cycle img").each(function(){
				var $slide = $(this);
				var ratio = $slide.width()/$slide.height();
				
				if( $slide.width() != $(".home-cycle").width() ) {
					$slide.width( $(".home-cycle").width() );
					$slide.height( $(".home-cycle").width() / ratio );
				}

				if( $slide.height() < $(".home-cycle").height() ) {
					$slide.height( $(".home-cycle").height() );
					$slide.width( $(".home-cycle").height() * ratio );
				}
				
				$slide.css({
					"margin-left": (($(".home-cycle").width() - $slide.width()) / 2) + "px",
					//"margin-top": (($(".home-cycle").height() - $slide.height()) / 2) + "px"
					"margin-top": 0 + "px"
				});
			});
		};
      $(window).resize(cycleresize);
      $(document).ready(cycleresize);
