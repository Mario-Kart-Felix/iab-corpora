var date = new Date();
var req;
var usingVehicleXML = true;
function loadXMLDoc(url) {
	if(url.match("vehicles.xml") == "vehicles.xml"){
		usingVehicleXML = true;	
	}else{
		usingVehicleXML = false;	
	}
	req = false;
    // branch for native XMLHttpRequest object
    if(window.XMLHttpRequest && !(window.ActiveXObject)) {
    	try {
			req = new XMLHttpRequest();
        } catch(e) {
			req = false;
        }
    // branch for IE/Windows ActiveX version
    } else if(window.ActiveXObject) {
       	try {
        	req = new ActiveXObject("Msxml2.XMLHTTP");
      	} catch(e) {
        	try {
          		req = new ActiveXObject("Microsoft.XMLHTTP");
        	} catch(e) {
          		req = false;
        	}
		}
    }
	if(req) {
		req.onreadystatechange = processReqChange;
		req.open("GET", url, true);
		req.send("");
	}
}

function processReqChange() {
    // only if req shows "loaded"
    if (req.readyState == 4) {
        // only if "OK"
        if (req.status == 200) {
            // ...processing statements go here...
			if(usingVehicleXML){
				useXML();
			}else{
				useVoiceXML();	
			}
        } else {
            alert("There was a problem retrieving the XML data:\n" + req.statusText);
        }
    }
}

var bundleName;
var vehicles;
var voices;
var description;
var displayName;
var codeName;
var vehicle_array = new Array();
var voice_array = new Array();

function useXML(){
		var breadCrumb = document.getElementById('breadCrumb');
		var bundleTitle = document.getElementById('bundleTitle');
		var descriptionArea = document.getElementById('descriptionArea');
 		bundleName = req.responseXML.getElementsByTagName(bundle)[0];
		vehicles = bundleName.getElementsByTagName("vehicle");
		voices = bundleName.getElementsByTagName("voice");
		
		displayName = bundleName.getAttribute("name");
		description = bundleName.getAttribute("description");
		codeName = bundleName.getAttribute("codeName");
		document.title = "Garmin | " + displayName;
		breadCrumb.innerHTML = displayName;
		bundleTitle.innerHTML = displayName;
		descriptionArea.innerHTML = description;
		
		var allVehicles = req.responseXML.getElementsByTagName("vehicles")[0];
		var everyVehicle = allVehicles.getElementsByTagName("vehicle");
		
		for(var n=0;n<vehicles.length;n++) {
			var vehicle = new Object();
			vehicle.vehicleName = getElementTextNS("", "vehicle", bundleName, n);
			for(var o=0; o<everyVehicle.length; o++){
				if(vehicle.vehicleName == everyVehicle[o].getAttribute("vehicleName")){
					//alert(everyVehicle[o].getAttribute("displayName"));
					vehicle.displayName = everyVehicle[o].getAttribute("displayName");
				}
			}
			vehicle_array.push(vehicle);
			
		}//end for loop
		
		if(voices.length > 0){
			loadXMLDoc('/vehicles/x/voices.xml?' + date.getTime());
			for(var p=0;p<voices.length;p++) {
				var voice = new Object();
				voice.voiceName = getElementTextNS("", "voice", bundleName, p);
				voice_array[p] = voice;
			}
		}else{
			createTable();	
			testForCommPlugin();
		}
}
function useVoiceXML(){
	var voiceGroup = req.responseXML.getElementsByTagName("voices")[0];
	var voiceElements = voiceGroup.getElementsByTagName("voice");
	
	for(var i=0;i<voice_array.length;i++) {
		//var vehicle = new Object();
		//vehicle.vehicleName = getElementTextNS("", "vehicle", bundleName, n);
		for(var j=0; j<voiceElements.length; j++){
			if(voice_array[i].voiceName == voiceElements[j].getAttribute("voiceName")){
				//alert(everyVehicle[o].getAttribute("displayName"));
				voice_array[i].displayName = voiceElements[j].getAttribute("displayName");
			}
		}
	}
	createTable();
	testForCommPlugin();
}
function getElementTextNS(prefix, local, parentElem, index) {
    var result = "";
    if (prefix && isIE) {
        // IE/Windows way of handling namespaces
        result = parentElem.getElementsByTagName(prefix + ":" + local)[index];
    } else {
        result = parentElem.getElementsByTagName(local)[index];
    }
    if (result) {
        // get text, accounting for possible
        // whitespace (carriage return) text nodes 
        if (result.childNodes.length > 1) {
            return result.childNodes[1].nodeValue;
        } else {
            return result.firstChild.nodeValue;    		
        }
    } else {
        return "n/a";
    }
}

		function xmlStringToObject(xmlData) {
			if (window.ActiveXObject) {
				//for IE
				xmlDoc=new ActiveXObject("Microsoft.XMLDOM");
				xmlDoc.async="false";
				xmlDoc.loadXML(xmlData);
				return xmlDoc;
			} else if (document.implementation && document.implementation.createDocument) {
				//for Mozilla
				parser=new DOMParser();
				xmlDoc=parser.parseFromString(xmlData,"text/xml");
				return xmlDoc;
			}
		}
