function urlencode(str)
{
str = (str+'').toString();
return encodeURIComponent(str).replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28'). replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/%20/g, '+');
}

function MakeURLSafe(val)
{
val = val.replace(/-/g, "%2d");
val = val.replace(/\+/g, "%2b");
val = val.replace(/\//g, "{47}");
//val = val.replace("/", "{47}");
val = urlencode(val);
val = val.replace(/\+/g, "-");
return val;
}

function getSearchOptions(searchtype,catuniversal) 
{
	var make = $('#searchmake').val();

	if(searchtype == 'make')
	{
		$('#searchmodel').find('option').remove().end();
		$('#searchmodel').append('<option value=\'\'>--select model--</option>');
		//return false;
	}

	var year = $('#searchyear').val();
	var universal = catuniversal;

	var params = 'autopopulate=true&make='+make+'&model='+model+'&year='+year+'&universal='+universal+'&ymmtype='+searchtype;

	$.ajax({
	type: "GET",

		var ymm_array=msg.split("~");
		if(searchtype == 'make')

			$('#searchmodel').find('option').remove().end();

				$('#searchyear').find('option').remove().end();
				$('#searchyear').append(ymm_array[1]);

			if(year == '')
			{
				$('#searchyear').find('option').remove().end();

			{
				$('#searchmodel').find('option').remove().end();
				$('#searchmodel').append(ymm_array[0]);
			}					
		}			   
	}
	});

}

function setYMM()
{
	var today = new Date();
	var expire = new Date();
	expire.setTime(expire.getTime()+(3600*24*1000*1000));

	var make = $('#searchmake').val();
	var model = $('#searchmodel').val();
	var year = $('#searchyear').val();

	var current_url = location.href;
	var search_url = location.href;
	var pos = current_url.indexOf("#");
	if(pos != -1)
	{
		current_url = current_url.substring(0,pos);
	}
	//current_url = current_url.replace(new RegExp(/#$/),"");
	current_url = current_url.replace(new RegExp(/\/$/),"");
	current_url = current_url.replace(new RegExp(/\/search\//),"/searchtext/");
	current_url = current_url.replace(new RegExp(/\/model_flag\/0/),"");
	current_url = current_url.split("/");
	var new_url = "";
	var xtraParam = "";

	for( var i =0; i <current_url.length; i++) 
	{
		if(current_url[i] == 'make' || current_url[i] == 'model' || current_url[i] == 'year' || current_url[i] == 'page')
		{
			i++;
		}
		else if(current_url[i] == 'searchtext')
		{
			if( typeof(SearchStringVal) != "undefined" )
			{
				new_url += "searchtext/"+SearchStringVal;
			}
			else if( search_url.indexOf("/search/") == -1 ) // this condition is added so that when searched and having no result, no need to retain search text.
			{
				new_url += "searchtext/"+current_url[eval(i+1)]+"/";
			}
				i++;	
		}
		else if(current_url[i] == 'orderby' || current_url[i] == 'sortby')
		{
				xtraParam += current_url[i]+"/"+current_url[eval(i+1)]+"/";
				i++;

                  <p class="center_text_normal"><strong>Truck Accessories and  SUV Accessories: An Overview and Some History</strong></p>
                  <p class="center_text_normal">Trucks and <a href="http://www.suv.org/">SUV</a>s  are of considerable variety and so are the accessories that enhance  them. Some people choose their truck or SUV for purely aesthetic  reasons, others because of specific performance needs. Which truck will  do a better job towing?&nbsp; Which SUV is the most <a href="http://www.epa.gov/greenvehicle/">fuel efficient</a>?  These are questions that the thoughtful consumer must consider before  choosing an SUV or truck. In this way, the consumer makes an informed  investment. But extracting the most from this investment requires  vigilant maintenance, wise fuel-consumption habits, and investing in <a href="http://www.bestinauto.com">performance accessories</a>. Examples of performance accessories are a <a href="http://www.bestinauto.com/magnaflow-exhausts.aspx">magnaflow exhaust</a> or an <a href="http://www.bestinauto.com/air-intakes.aspx">air intake</a>. The last bit is what our  site is devoted to, the SUV and <a href="http://en.wikipedia.org/wiki/Truck_Accessories">Truck Accessory</a>.  Of course, we also sell accessories for enhancing the visual appeal of  your vehicle. If you're going to turn your truck or SUV into a  high-performance machine, you ought to make sure it looks like one!&nbsp; It  is interesting to note that such aesthetic enhancements are accounting  for more and more of the specialty auto accessory market share.  According to <a href="http://www.sema.org/main/SemaOrgHome.aspx?ID=/content/SEMAorg/HomePage">SEMA</a> (Speciality Equipment and Marketing Association), the &quot;accessories and  appearance&quot; category was the only one to increase its share of the  &ldquo;specialty equipment&rdquo; market over the last ten years, rising from 51.1%  in 1996 to 58.1% in 2005.</p>

                  <p class="center_text_normal">Much like the vehicles which they modify, specialty performance  equipment and truck accessories enjoy an interesting history.&nbsp;  According to the <a href="http://www.sema.org">SEMA</a>, the truck  accessory industry finds its beginning in the 1930&rsquo;s. Once motor  vehicles became available to a broad base of consumers, a group of  enthusiasts (mostly interested in racing and competing in other ways  with fellow motorists) quickly emerged. The years leading up to <a href="http://www.bbc.co.uk/history/worldwars/wwtwo/">World  War II</a> saw racers in the southern Californian deserts forming their own  &ldquo;R&amp;D&rdquo; (i.e. Research and Development) organizations as they  tinkered with stock equipment and made all manners of ad hoc  modifications. As with so many other domestic industries, the expansion  and standardization of auto accessories was delayed by World War II,  which placed unprecedented demands on U.S. resources.&nbsp; Following the  war, however, the new expertise acquired from the maintenance of  military trucks, coupled with the widespread war-starved consumer  desire for new vehicles and innovation, gave birth to a full-fledged  truck accessories industry, the sole purpose of which would be to give  consumers the freedom to augment/enhance their vehicles as they saw fit  (within the limits of the law, of course).&nbsp; The manufacturers you will  find at TruckChamp.com, from Bully Dog (performance diesel products) to  Westin, Inc. (makers of Westin Nerf Bars), can all claim some share in  the history we have briefly touched upon here. Read the entire Truck  Champ <a href="/pages/history-of-truck-accessories">Truck Accessories Overview and Industry Information</a> Article... </p>

                  <p class="center_text_normal"><strong>A Brief History of Pickup Trucks and SUVs</strong></p>
                  <p class="center_text_normal"> There are certain artifacts of Americana which, aside from possessing a  fascinating history in their own right, offer a unique vantage point  from which their broader cultural milieu can be understood. The pickup  truck and the SUV are examples of such.&nbsp; Their presence has been felt  in almost every aspect of <a href="http://www.usa.gov/Citizen/Topics/History_Culture.shtml">American culture</a>,  from song lyrics to visual art, and even in politics. Who could forget,  for instance, when during the 2004 race Senator Fred Thompson  (R-Tenn.)&nbsp; summed-up the faults of his opponent with the assertion, &ldquo;He  hasn't spent enough time in a pickup truck.&quot;? (For a more extensive  treatment of the cultural significance, see this <a href="http://en.wikipedia.org/wiki/Pickup_truck">Wikipedia article on Pickup Trucks.</a></p>
                  <p class="center_text_normal">Like many other staples of American culture, e.g.  rock-n-roll and pizza, the pickup truck finds its roots on another  shore. Accordingly, our all-too-brief &quot;off-road&quot; journey through its  history shall begin in Australia, where the &ldquo;ute&rdquo; (i.e. utility truck)  first appeared, and culminate in an introduction to the world of truck  accessories and SUV accessories; additions to your vehicle that will  enhance its performance, appearance, and longevity...just like a <a href="http://www.bestinauto.com/hitches.aspx">trailer hitch</a>. Read the entire  Truck Champ <a href="/pages/history-of-trucks-suvs">History of Pickups and SUVs</a>...</p>
