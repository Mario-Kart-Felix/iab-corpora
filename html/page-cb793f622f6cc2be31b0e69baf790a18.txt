		 function validateEmail(inEmail)
		{
			var atPtrn;
			var dotPtrn;
			var iAtCount;
			var iWhitespaceCount;
			var regObj;
			var bAt;
			var bDot;
		
			atPtrn = "@";
			dotPtrn = ".";
			iAtCount = 0;
			iWhitespaceCount = 0;
	
			regObj = new RegExp(atPtrn);
			bAt = regObj.test(inEmail);
	
			regObj = new RegExp(dotPtrn);
			bDot = regObj.test(inEmail);
	
			if(!bDot || !bAt)
			{
				return false;
			}
			
			for(var i = 0; i < inEmail.length; i++)
			{
				if(inEmail.charCodeAt(i) == 64)
				{
					iAtCount++;
				}
				if(inEmail.charCodeAt(i) == 32)
				{
					iWhitespaceCount++;
				}		
			}
	
			if(iAtCount > 1)
			{
				return false;
			}
	
			if(iWhitespaceCount > 0)
			{
				return false;
			}
			
			if((inEmail.length - inEmail.lastIndexOf(".")) > 4)
			{
				return false;
			}
		
			return true;
	
		}	//end function validateEmail

		function formatPhone(inObj)
		{
			var t;
			var returnT;
		
			t = inObj.value;	
		
			//if left blank skip formatting
			if( t.length > 0 ) 
			{
	  			if( t.length == 10 )
				{
					//user entered # in 9999999999 format
					if(!isNaN(t))
					{			   
						//characters are all #'s format to (999)999-9999 format
						returnT = aPF(t);
						inObj.value = returnT;
					}
				}
				else if(t.length == 12)
				{
					//inspect other scenarios for phone # potentials
					returnT = oPFormats(t);
					inObj.value = returnT;
				}
				else if(t.length == 11)
				{
					//user entered in long distance number
					if(!isNaN(t))
					{
						//if all #'s then we can format
						returnT = aLDPFormat(t);
						inObj.value = returnT;
					}
				}
			
			}	//end if for length check
	
		} //end formatPhone function


		function aLDPFormat(inval)
		{
			var phonenum;
	
			phonenum = "";
	
			phonenum += inval.slice(0, 1) + "-" + inval.slice(1, 4);
			phonenum += "-" + inval.slice(4, 7) + "-" + inval.slice(7);
	
			return phonenum;
		}

		function oPFormats(inval)
		{
			//got the 12 character format entry ( 999.999.9999 ) ( "."'s could be anything such as "-"'s  etc...
	
			var phonenum;
	
			phonenum = "";	
			phonenum += "(" + inval.slice(0, 3) + ")";	
			phonenum += inval.slice(4, 7) + "-";
			phonenum += inval.slice(8);
	
			return phonenum;
		}


		function aPF(inval)

			phonenum += inval.slice(0, 3) + ")";
			phonenum += inval.slice(3, 6) + "-";
			phonenum += inval.slice(6);
	
			return phonenum;
	
		}	//end function aFF
