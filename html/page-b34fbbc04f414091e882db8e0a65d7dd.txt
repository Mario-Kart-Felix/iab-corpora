    function swapStates() {
        var countrySelect = document.frmSearch.country;
        var stateSelect = document.frmSearch.state;
        var countrySelectedIndex = countrySelect.selectedIndex;
        
        stateSelect.options.length = 0;
        stateSelect.disabled=true;
        
        if (states[countrySelectedIndex] != null)
        {
            var defaultOption = "All states";
            if (countrySelectedIndex == 2) defaultOption = "All province";
            stateSelect.options[0] = new Option(defaultOption,"");
            
            for (var i=0; i < states[countrySelectedIndex].length; i++)
            {
                stateSelect.options[i+1] = new Option(states[countrySelectedIndex][i][1],states[countrySelectedIndex][i][0]);
            }
            
            stateSelect.disabled=false;
        }
    }
