}

newParams[k] = response.SBdefaults[k] || 'clear';
} else if (response.defaults) {
newParams[k] = response.defaults[k.replace('bodyStyle', 'bodystyle')] || 'clear';
}
}
// if for some reason selectedIndex is out of range, use first index (0)
el.selectedIndex = (selectedIndex < el.options.length) ? selectedIndex : 0;
// if there are no options in this set, disable the select element
if (itemCount === 0) {
$(el).addClass('disabled').attr('disabled', 'disabled');
} else {
$(el).removeClass('disabled');
}
}
}
}
this.setParams(newParams);
if (trueInventoryType) {
this.setInventoryType(trueInventoryType);
}
$('form', '#ajaxSmartbrowse').attr('action', smartbrowse.getInventoryType().inv);
endLoading();
}
