function getCity() {

		url: '/ajx_city.php',

$("#cityd ul li a").hover(
   function () {
      currentSelection = $(this).data("number");
      setSelected(currentSelection);
   }, function() {
      $("#cityd ul li a").removeClass("itemhover");
      currentUrl = '';
   }
);
 
 function setSelected(menuitem) {
   $("#cityd ul li a").removeClass("itemhover");
   $("#cityd ul li a").eq(menuitem).addClass("itemhover");
   currentUrl = $("#cityd ul li a").eq(menuitem).attr("city");
}

 function navigate(direction) {
   // Check if any of the menu items is selected
   if($("#cityd ul li .itemhover").size() == 0) {
      currentSelection = -1;
   }
   
   if(direction == 'up' && currentSelection != -1) {
      if(currentSelection != 0) {
         currentSelection--;
      }
   } else if (direction == 'down') {
      if(currentSelection != $("#cityd ul li").size() -1) {
         currentSelection++;
      }
   }
   setSelected(currentSelection);
}


 $(document).keydown(function(e) {

         break;

   });
