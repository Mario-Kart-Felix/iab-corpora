            directionsDisplay: new google.maps.DirectionsRenderer(),
            directionsService: new google.maps.DirectionsService(),

            initialize: function(latlng) {
                var options = {
                  zoom: 13,
                  center: latlng,
                  scrollwheel: false,
                  mapTypeId: google.maps.MapTypeId.ROADMAP
                };

                this.map = new google.maps.Map(document.getElementById("map_canvas"), options);
                if(latlng) {
                    this.marker = new google.maps.Marker({position: latlng});
                    this.marker.setMap(this.map);
                }
                this.directionsDisplay.setMap(this.map);
            },

            getDirection: function(fromAddress, toAddress, locale) {
                if(document.getElementById("directions")) {
                    var divmap = document.getElementById("directions");
                    divmap.innerHTML = '';
                }

                var request = {
                    origin:fromAddress,
                    destination:toAddress,
                    travelMode: google.maps.DirectionsTravelMode.DRIVING
                };

                this.directionsService.route(request, function(response, status) {
                  if (status == google.maps.DirectionsStatus.OK) {
                    if(map_and_direction.marker) {
