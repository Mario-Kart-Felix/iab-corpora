 	function get_quote() {
 		var storingZip = document.getElementById('hdnStoringZip')
 		var buyerZip = document.getElementById('buyerZip')
 		var vehicleYear = document.getElementById('hdnVehicleYear')
 		var vehicleMake = document.getElementById('hdnVehicleMake')
 		var vehicleModel = document.getElementById('hdnVehicleModel')
 		var hdnSendQuoteEmail = document.getElementById('hdnSendQuoteEmail');

 		var quoteID = "";
 		var totalPrice = document.getElementById('shipquoteprice');
 		var quoteReturnURL = "";
 		var resultstext = document.getElementById('shipresultstext');
 		var shippingButton = document.getElementById('shippingButton');
 		var shiptopline = document.getElementById('shiptopline');
 		var divshiptoLink = document.getElementById('divshiptoLink');
 		var shiptoLink = document.getElementById('shiptoLink');
 		var shipquoteloader = document.getElementById('shipquoteloader');
 		var shippingform = document.getElementById('shipping');


 		var hdnQuoteID = document.getElementById('hdnQuoteID');

 		var hdnImageUrl = document.getElementById('hdnImageUrl');

 		if (!buyerZip) {
 			divMessages.innerHTML = "Sorry, there was an issue with your zip code";
 			shippingButton.value = "Let's try this again";
 			shippingButton.onclick = new Function("try_again('get')");
 			shiptopline.style.display = 'none';
 			divshiptoLink.style.display = 'none';
 			buyerZip.style.display = 'none';
 			divMessages.style.display = 'block';
 			return false;
 		}
 		else {
 			if ((buyerZip.value == '') || (!ValidCSZip(buyerZip.value))) {
 				divMessages.innerHTML = "Sorry, there was an issue with your zip code";
 				shippingButton.value = "Let's try this again";
 				shippingButton.onclick = new Function("try_again('get')");
 				shiptopline.style.display = 'none';
 				divshiptoLink.style.display = 'none';
 				buyerZip.style.display = 'none';
 				divMessages.style.display = 'block';
 				return false;
 			}
 		}

 		shippingform.style.display = 'none';
 		shipquoteloader.style.display = 'block';

 		try {
 			CarSoup.xhrGet({
 				url: "/JsonResult.aspx?fn=GetShippingQuote&storingZip=" + storingZip.value + "&buyerZip=" + buyerZip.value + "&vehicleYear=" + vehicleYear.value + "&vehicleMake=" + vehicleMake.value + "&vehicleModel=" + vehicleModel.value,
 				load: function (response) {
 					if (response) {

 						shippingform.style.display = 'block';
 						shipquoteloader.style.display = 'none';

						var results = JSON.parse(response)
						if (results.success) {
							if (results.isError) {
								buyerZip.style.display = 'none';
								divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
								divMessages.style.display = 'block';
								shippingButton.onclick = new Function("try_again('get')");
								shippingButton.value = "Let's try this again";
							}
							else {
								shiptopline.style.display = 'inline-block';
								divshiptoLink.style.display = 'inline-block';
								totalPrice.innerHTML = "$" + results.totalPrice;
								totalPrice.style.display = 'block';
								shippingButton.value = "Email Quote";
								buyerZip.value = "";
								buyerZip.placeholder = "Enter your email address";
								shippingButton.onclick = new Function("send_quote()");
								hdnQuoteID.value = results.quoteID;
								shiptoLink.href = results.quoteReturnURL + hdnImageUrl.value;
								hdnSendQuoteEmail.value = results.quoteReturnURL;
								shippingform.onsubmit = new Function("send_quote(); return false;");
							}			

						}
						else {

							shippingform.style.display = 'block';
							shipquoteloader.style.display = 'none';

							if (results.isError) {
								buyerZip.style.display = 'none';
								divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
								divMessages.style.display = 'block';
								shippingButton.onclick = new Function("try_again('get')");
								shippingButton.value = "Let's try this again";
							}
							else {
								buyerZip.style.display = 'none';
								divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
								divMessages.style.display = 'block';
								shippingButton.onclick = new Function("try_again('get')");
								shippingButton.value = "Let's try this again";
							}
						}

 					}
 				},
 				error: function (response) {

 					shippingform.style.display = 'block';
 					shipquoteloader.style.display = 'none';

 					buyerZip.style.display = 'none';
 					divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
 					divMessages.style.display = 'block';
 					shippingButton.onclick = new Function("try_again('get')");

 			buyerZip.style.display = 'none';

 			shippingButton.onclick = new Function("try_again('get')");
 			shippingButton.value = "Let's try this again";
 			return false;
 		}
 		return false;
 	}

 	function send_quote() {
 		var quoteID = document.getElementById('hdnQuoteID')
 		var email = document.getElementById('buyerZip')
 		var resultstext = document.getElementById('shipresultstext')
 		var shiptopline = document.getElementById('shiptopline');
 		var divshiptoLink = document.getElementById('divshiptoLink');
 		var divMessages = document.getElementById('divMessages');
 		var buyerZip = document.getElementById('buyerZip');
 		var hdnSendQuoteEmail = document.getElementById('hdnSendQuoteEmail');
 		var hdnImageUrl = document.getElementById('hdnImageUrl');
 		var shipquoteloader = document.getElementById('shipquoteloader');
 		var shippingform = document.getElementById('shipping');
 		var shippingButton = document.getElementById('shippingButton');

 		var loadermessage = document.getElementById('loadermessage');
 		loadermessage.innerHTML = "Sending Quote...";

 		var reEmail = /^(([A-Za-z0-9]+_+)|([A-Za-z0-9]+\-+)|([A-Za-z0-9]+\.+)|([A-Za-z0-9]+\++))*[A-Za-z0-9]+@((\w+\-+)|(\w+\.))*\w{1,63}\.[a-zA-Z]{2,6}$/;
 		if ((!reEmail.test(email.value)) || (email.value == '' || email == 'enter email address')) {
 			divMessages.innerHTML = "Sorry, there was an issue with your email address";
 			shippingButton.value = "Let's try this again";
 			shippingButton.onclick = new Function("try_again('send')");
 			shiptopline.style.display = 'none';
 			divshiptoLink.style.display = 'none';
 			buyerZip.style.display = 'none';
 			divMessages.style.display = 'block';
 			return false;
 		}

 		shippingform.style.display = 'none';
 		shipquoteloader.style.display = 'block';

 		try {
 			CarSoup.xhrGet({
 				url: "/JsonResult.aspx?fn=SendShippingQuote&quoteID=" + quoteID.value + "&email=" + email.value,
 				load: function (response) {
 					if (response) {

 						shippingform.style.display = 'block';
 						shipquoteloader.style.display = 'none';

 						var results = JSON.parse(response)
 						if (results.success) {							
 								divMessages.innerHTML = "Your quote has been sent!";
 								shippingButton.value = "Learn More";
 								shiptopline.style.display = 'none';
 								divshiptoLink.style.display = 'none';
 								buyerZip.style.display = 'none';
 								divMessages.style.display = 'block';
 								shippingButton.onclick = new Function("learn_more('" + hdnSendQuoteEmail.value + hdnImageUrl.value + "')");
 						}
 						else {
 							buyerZip.style.display = 'none';
 							divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
 							divMessages.style.display = 'inline-block';
 							shippingButton.onclick = new Function("try_again('send')");
 							shippingButton.value = "Let's try this again";
 						}

 					}
 				},
 				error: function (response) {
 					shippingform.style.display = 'block';
 					shipquoteloader.style.display = 'none';

 					buyerZip.style.display = 'none';
 					divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
 					divMessages.style.display = 'block';
 					shippingButton.onclick = new Function("try_again('send')");
 					shippingButton.value = "Let's try this again";
 				}
 			});
 		}
 		catch (e) {
 			shippingform.style.display = 'block';
 			shipquoteloader.style.display = 'none';

 			buyerZip.style.display = 'none';
 			divMessages.innerHTML = "Sorry, your quote couldn't be calculated";
 			divMessages.style.display = 'block';
 			shippingButton.onclick = new Function("try_again('send')");
 			shippingButton.value = "Let's try this again";
 		
 			return false;
 		}

 		return false;
 	}

 	function try_again(mode) {
 		var divMessages = document.getElementById('divMessages');
 		var buyerZip = document.getElementById('buyerZip');
 		var shippingButton = document.getElementById('shippingButton');
 		var shiptopline = document.getElementById('shiptopline');
 		var divshiptoLink = document.getElementById('divshiptoLink');
 		var totalPrice = document.getElementById('shipquoteprice');
 		var shippingform = document.getElementById('shipping');

 		buyerZip.value = "";

 		if (mode == "get") {
 			shiptopline.style.display = 'none';
 			divshiptoLink.style.display = 'none';
 			shippingButton.onclick = new Function("get_quote()");
 			shippingButton.value = "Get Quote";
 			shippingform.onsubmit = new Function ("get_quote(); return false;");
 			buyerZip.placeholder = "Enter your zip code to get a quote";
 		}
 		else {
 			shippingButton.onclick = new Function("send_quote()");
 			shippingButton.value = "Email Quote";
 			shippingform.onsubmit = new Function ("send_quote(); return false;");
 			buyerZip.placeholder = "Enter your email address";
 			shiptopline.style.display = 'inline-block';
 			divshiptoLink.style.display = 'inline-block';
 			totalPrice.style.display = 'block';
 		}
 		buyerZip.style.display = 'block';
 		divMessages.style.display = 'none';
 		
 	}
