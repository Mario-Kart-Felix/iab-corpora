                            var temp = $.trim(value).split(/\s+/);
                            return [temp.slice(1, 13)];
                        });

                        var lastValue = parseFloat(cpiTable[cpiTable.length - 1].slice(-1)[0]);

                        $('#priceAdjusted').html((function () {
                            var dateArray = $('#dateSold').html().split(/\//);
                            if(dateArray.length === 3) {
                                var row = parseInt(dateArray[2]) - firstYear;
                                var column = parseInt(dateArray[0].replace(/^0/, '')) - 1;
                                var soldDateInflationMultiplier = cpiTable[row][column];

                                var tempPrice = (parseFloat($pn.html().replace(/\W+/, '').replace(',', '')) * (1 + ((lastValue - soldDateInflationMultiplier) / lastValue)));

                                return '$' + tempPrice.toString().match(/\d+\.\d{2}/);

                            }
                        })());

                        $('#priceAdjustedContainer').css('visibility', 'visible');
                    });
        }
        else {
            // No point to use up the space for non-US dollar countries
            $('#priceAdjustedContainer').hide();
        }
    });
</script>
