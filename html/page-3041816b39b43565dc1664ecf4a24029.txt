function urlencode(str)
{
str = (str+'').toString();
return encodeURIComponent(str).replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28'). replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/%20/g, '+');
}

function MakeURLSafe(val)
{
val = val.replace(/-/g, "%2d");
val = val.replace(/\+/g, "%2b");
val = val.replace(/\//g, "{47}");
//val = val.replace("/", "{47}");
val = urlencode(val);
val = val.replace(/\+/g, "-");
return val;
}

function getSearchOptions(searchtype,catuniversal) 
{
	var make = $('#searchmake').val();

	{
		$('#searchmodel').find('option').remove().end();
		$('#searchmodel').append('<option value=\'\'>--select model--</option>');
		//return false;
	}

	var year = $('#searchyear').val();

	var params = 'autopopulate=true&make='+make+'&model='+model+'&year='+year+'&universal='+universal+'&ymmtype='+searchtype;

			if(year == '')

				$('#searchyear').find('option').remove().end();

		}			   
	}
	});

}

function setYMM()
{
	var today = new Date();
	var expire = new Date();
	expire.setTime(expire.getTime()+(3600*24*1000*1000));

	var make = $('#searchmake').val();
	var model = $('#searchmodel').val();
	var year = $('#searchyear').val();

	var current_url = location.href;
	var search_url = location.href;
	var pos = current_url.indexOf("#");
	if(pos != -1)
	{
		current_url = current_url.substring(0,pos);
	}
	//current_url = current_url.replace(new RegExp(/#$/),"");
	current_url = current_url.replace(new RegExp(/\/$/),"");
	current_url = current_url.replace(new RegExp(/\/search\//),"/searchtext/");
	current_url = current_url.replace(new RegExp(/\/model_flag\/0/),"");
	current_url = current_url.split("/");
	var new_url = "";
	var xtraParam = "";

	for( var i =0; i <current_url.length; i++) 
	{
		if(current_url[i] == 'make' || current_url[i] == 'model' || current_url[i] == 'year' || current_url[i] == 'page')
		{
			i++;
		}
		else if(current_url[i] == 'searchtext')
		{
			if( typeof(SearchStringVal) != "undefined" )
			{
				new_url += "searchtext/"+SearchStringVal;
			}
			else if( search_url.indexOf("/search/") == -1 ) // this condition is added so that when searched and having no result, no need to retain search text.
			{
				new_url += "searchtext/"+current_url[eval(i+1)]+"/";
			}
				i++;	
		}
		else if(current_url[i] == 'orderby' || current_url[i] == 'sortby')
		{
				xtraParam += current_url[i]+"/"+current_url[eval(i+1)]+"/";
				i++;
		}
		else
		{
			new_url += current_url[i]+"/";
		}
	}

	if( typeof(SearchQualVal) != "undefined" )
	{
		new_url +=	SearchQualVal;

{
str = (str+'').toString();
return encodeURIComponent(str).replace(/!/g, '%21').replace(/'/g, '%27').replace(/\(/g, '%28'). replace(/\)/g, '%29').replace(/\*/g, '%2A').replace(/%20/g, '+');
}

function setYMM()
{
	var today = new Date();
	var expire = new Date();
	expire.setTime(expire.getTime()+(3600*24*1000*1000));

	var make = $('#searchmake').val();
	var model = $('#searchmodel').val();
	var year = $('#searchyear').val();

	var current_url = location.href;
	var search_url = location.href;
	var pos = current_url.indexOf("#");
	if(pos != -1)
	{
		current_url = current_url.substring(0,pos);
	}
	//current_url = current_url.replace(new RegExp(/#$/),"");
	current_url = current_url.replace(new RegExp(/\/$/),"");
	current_url = current_url.replace(new RegExp(/\/search\//),"/searchtext/");
	current_url = current_url.replace(new RegExp(/\/model_flag\/0/),"");
	current_url = current_url.split("/");
	var new_url = "";
	var xtraParam = "";

	for( var i =0; i <current_url.length; i++) 
	{
		if(current_url[i] == 'make' || current_url[i] == 'model' || current_url[i] == 'year' || current_url[i] == 'page')
		{
			i++;
		}
		else if(current_url[i] == 'searchtext')
		{
			if( typeof(SearchStringVal) != "undefined" )
			{
				new_url += "searchtext/"+SearchStringVal;
			}
			else if( search_url.indexOf("/search/") == -1 ) // this condition is added so that when searched and having no result, no need to retain search text.
			{
				new_url += "searchtext/"+current_url[eval(i+1)]+"/";
			}
				i++;	
		}
		else if(current_url[i] == 'orderby' || current_url[i] == 'sortby')
		{
				xtraParam += current_url[i]+"/"+current_url[eval(i+1)]+"/";
				i++;
		}
		else
		{
			new_url += current_url[i]+"/";
		}
	}

	if( typeof(SearchQualVal) != "undefined" )
	{
		new_url +=	SearchQualVal;
	}

	if(make != '')
		new_url += "make/"+encodeURIComponent(make)+"/";

//	if(model != '')
//		new_url += "model/"+encodeURIComponent(model)+"/";

	if(year != '')
		new_url += "year/"+year+"/";

	if(xtraParam != "")
		new_url += xtraParam;

	new_url += "#focusDiv";
	
	if(year == '')
	{
		document.cookie = 'last_search_selection[year]=; expires=; path=/;';
	}else{
		document.cookie = 'last_search_selection[year]='+year+'; expires='+expire.toGMTString()+'; path=/;';
	}
		
	if(make == '')
	{
		document.cookie = 'last_search_selection[make]=; expires=; path=/;';
	}else{
		document.cookie = 'last_search_selection[make]='+decodeURI(make)+'; expires='+expire.toGMTString()+'; path=/;';
	}
		
	if(model == '')
	{
		document.cookie = 'last_search_selection[model]=; expires=; path=/;';
	}else{
		document.cookie = 'last_search_selection[model]='+decodeURI(model)+'; expires='+expire.toGMTString()+'; path=/;';
	}

	location.href=new_url;

}

<h3 style="color: #990000; font-size: 1.5em; line-height: 1.2em;">Ford Motor Company</h3>
<p>First incorporated in 1903, Ford has long been a household name in America. Chances are, if you don't own one, your brother, father, cousin, grandfather or Great Aunt Matilda did. The company is currently the second largest automaker in the U.S. and is the fourth-largest in the world based on the number of vehicles sold annually.</p>
<p>Many of our generation's most popular, well-known vehicles were manufactured by Ford. Remember the Gran Torino? And who doesn't remember the original Mustang? Ford first introduced the Mustang in 1964 at th New York World's Fair - the company's most successful launch since the Model A. The car inspired a "pony car class" of American automobiles to competitors, such as AMC's Javelin, Chrysler's revamped Plymouth Barracuda and GM's Chevy Camaro. And while some other pony cars have seen a revival, only Ford's Mustang has been in production for four uninterrupted decades.</p>
