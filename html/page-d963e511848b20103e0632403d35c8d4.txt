				$('#contact-form').hide();
			}
			else {
				$('#contact-form input, #contact-form textarea').removeClass('error');
				for (i=0; i<data.error_field.length; i++)
				{
					$('#contact-form #'+data.error_field[i]).addClass('error');
				}
			}
