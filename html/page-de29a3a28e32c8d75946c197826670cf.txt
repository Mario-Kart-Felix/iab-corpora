function mailTest(emailAddress) {
	errorMessage = "";
	if (emailAddress.length > 0) {
		re = /^([\w-]+\.?)*\w+@([\da-zA-z-]+\.)+[a-zA-z]{2,3}$/
		if (!re.test(emailAddress)) {
			//email address did not match our regexp, so we need to dig deeper to find the problem
			if (emailAddress.search('@') == -1) {
				errorMessage = "No @ sign detected.  An @ sign is part of every e-mail address.\n";
			} else if (emailAddress.charAt(emailAddress.length - 1) == "@") {
				errorMessage = "An @ sign cannot be the last character of the e-mail address.\n";
			} else if ((emailAddress.charAt(0) == "@")) {
				errorMessage = "An @ sign cannot be the first character of the e-mail address.\n";
			} else if ( emailAddress.indexOf("@") != emailAddress.lastIndexOf("@") ) {
				errorMessage = "A valid e-mail address contains only one @ sign.\n";
			} else if (emailAddress.search('\\.') == -1) {
				errorMessage = "No period detected.  An e-mail address contains at least one period.\n";
			} else if (emailAddress.charAt(emailAddress.length - 1) == ".") {
				errorMessage = "The last character of the e-mail address cannot be a period.\n";		
			} else if (emailAddress.charAt(0) == ".") {
				errorMessage = "The first character of the e-mail address cannot be a period.\n";		
			} else if (emailAddress.search("\\,") > -1) {
				errorMessage = "A valid e-mail address cannot contain a comma.  If you have a Compuserve account, substitute a period for the comma in your Compuserve ID, like so: 12345.6789@compuserve.com.\n";		
			} else if (emailAddress.search(" ") > -1) {
				errorMessage = "You cannot have a space as part of a single e-mail address.\n";		
			} else if (emailAddress.search("\\*") > -1) {
				errorMessage = "You cannot have an asterisk in an e-mail address.\n";		
			} else if (emailAddress.search("\\(") > -1) {
				errorMessage = "You cannot have a open parentheses in an e-mail address.\n";		
			} else if (emailAddress.search("\\)") > -1) {
				errorMessage = "You cannot have a close parentheses in an e-mail address.\n";		
			} else if (emailAddress.search(">") > -1) {
				errorMessage = "You cannot have a > in an e-mail address.\n";		
			} else if (emailAddress.search("<") > -1) {
				errorMessage = "You cannot have a < in an e-mail address.\n";		
			} else if (emailAddress.search(":") > -1) {
				errorMessage = "You cannot have a colon in an e-mail address.\n";		
			} else if (emailAddress.search(";") > -1) {
				errorMessage = "You cannot have a semicolon in an e-mail address.\n";		
			} else if (emailAddress.search("\"") > -1) {
				errorMessage = "You cannot have a double quote in an e-mail address.\n";		
			} else if (emailAddress.search("\\'") > -1) {
				errorMessage = "You cannot have a single quote in an e-mail address.\n";		
			} else if (emailAddress.search("\\.\\.") > -1) {
				errorMessage = "You cannot have two consecutive periods in an e-mail address.\n";		
			} else if (emailAddress.search("@\\.") > -1) {
				errorMessage = "An @ sign cannot be directly followed by a period.\n";		
			} else if (emailAddress.search("\\.@") > -1) {
				errorMessage = "An @ sign cannot be directly preceded by a period.\n";		
			}
		}
	} else {
		//error created because email address is empty.
		errorMessage = "You must provide an email address.\n";		
	}
	return errorMessage;
}
