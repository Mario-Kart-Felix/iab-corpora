var g_map = null;
var g_marker = null;
var g_infoWindow = null;
var g_address_attempt = 1;

function initializeMap() {

	var geocodeOptions = {       
		 address: '17871+Sampson+Lane+Huntington+Beach+CA+92647'
	};     
	var geocoder = new google.maps.Geocoder();
	geocoder.geocode(geocodeOptions, geocodeCallback);
} // initializeMap

function geocodeCallback(results, status) {
//	alert(status);
	if (status == google.maps.GeocoderStatus.OK) {
		var latlng = results[0].geometry.location;

		var mapOptions = {       
			 zoom: 13
			,center: latlng
			,mapTypeId: google.maps.MapTypeId.ROADMAP
			,mapTypeControl: false
		};     
		g_map = new google.maps.Map(document.getElementById("map"), mapOptions);
	
		var markOptions = {
			 map: g_map
			,position: latlng
			,draggable: true
			,flat: true
		};
		g_marker = new google.maps.Marker(markOptions);
	
		var infoWindowOptions = {       
			 maxWidth: 220
			,content: '<b>Orange County Offroad & Auto Repair</b><br>' + '17871 Sampson Lane<br>Huntington Beach, CA 92647'
		};
		g_infoWindow = new google.maps.InfoWindow(infoWindowOptions);
		window.setTimeout('showInfoWindow();', 1000); // initially show the info window.  Have to setTimeout so the map will slide to adjust for the info window.
		google.maps.event.addListener(g_marker, 'click', showInfoWindow); // re-show info window when clicked.

		g_map.setCenter(results[0].geometry.location);
	}
	else if (status == google.maps.GeocoderStatus.ZERO_RESULTS) {
		if (g_address_attempt < 2) {
			g_address_attempt++;

			var geocodeOptions = {       
				 address: 'Huntington Beach CA 92647'
			};
			var geocoder = new google.maps.Geocoder();
			geocoder.geocode(geocodeOptions, geocodeCallback);
		}
	}
	else {
		// some other problem.
//		alert("Geocode was not successful for the following reason: " + status);
	}
} // geocodeCallback


function showInfoWindow() {
	g_infoWindow.open(g_map, g_marker);
} // showInfoWindow
