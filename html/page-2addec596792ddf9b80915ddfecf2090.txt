    function onVideoTemplateReady(inEvent) {
        var theExperienceModule = itsPlayer.getModule(APIModules.EXPERIENCE);
        theExperienceModule.removeEventListener(BCExperienceEvent.TEMPLATE_READY, onVideoTemplateReady);
        cueTheVideo(thePlayerModule.getCurrentVideo().id);
    }

    function cueTheVideo(inVideoId) {
        renderOverlay();
        itsPlayer.play();
    }


    onMediaProgress = function(event) {
        var itsVideoPlayer = itsPlayer.getModule(APIModules.VIDEO_PLAYER);
        var currentVideo = itsVideoPlayer.getCurrentVideo();
        var overlayprops;
        if(currentVideo.customFields.overlayprops != undefined){
            overlayprops = eval(currentVideo.customFields.overlayprops);
        }
        //overlayprops = eval("[{image: 'http://www.consumerreports.org/etc/designs/video/overlay/horiz_overlay_black_road_test.png', url:'http://www.consumerreports.org/cro/subaru/legacy/photos-and-video.htm?tab=video#photosVideosTab',start: '00:12:00',end: '00:17:00'},{image: 'http://www.consumerreports.org/etc/designs/video/overlay/horiz_overlay_white_road_test.png', url:'http://www.consumerreports.org/cro/subaru/legacy/photos-and-video.htm?tab=video#photosVideosTab',start: '01:27:00',end: '01:31:00'}]");
        if(overlayprops != undefined && overlayprops.length > 0){
            var displayOverlay = false;
            var image;
            var url;

            for(var i=0; i<overlayprops.length; i++){
                var overlaystarttime = overlayprops[i].start;
                var overlayendtime = overlayprops[i].end;
                url = overlayprops[i].url;
                image = overlayprops[i].image;

                if(overlaystarttime != undefined && overlayendtime != undefined){
                    var startMin  = overlaystarttime.substring(0, 2);
                    var startSecs  = overlaystarttime.substring(3, 5);
                    var endMin  = overlayendtime.substring(0, 2);
                    var endSecs  = overlayendtime.substring(3, 5);
                    var start = startMin*60 + startSecs*1;
                    var end = endMin*60 + endSecs*1;

                    if(start <= end && event.position >= start && event.position <= end) {
                        displayOverlay = true;
                        break;
                    }
                }
            }

            if(!displayOverlay){
                itsPlayer_overlayImage.setVisible(false);
            }else{
                itsPlayer_overlayImage.setSource(image);
                //itsPlayer_overlayImage.setURL(url);

                itsPlayer_overlayImage.setVisible(true);
                videoUrl = url;
            }
        }else{
            itsPlayer_overlayImage.setVisible(false);
        }
    }

        //itsPlayer_overlayImage.setVisible(true);
    }

    function renderOverlay(){

        itsPlayer_layout.appendChild("<Image id=\"overlayImage\" y=\"12\" x=\"12\" height=\"90\" width=\"216\" visible=\"true\"/>");
        itsPlayer_overlayImage = itsPlayer_layout.getChildAt(1);

        itsPlayer_videoPlayer.addEventListener(BCMediaEvent.PROGRESS, onMediaProgress);
        itsPlayer_videoPlayer.addEventListener(BCMediaEvent.STOP, onMediaStop);
        itsPlayer_overlayImage.addEventListener("elementClick", onLinkClick);
    }

    function onLinkClick(event){
        window.location.href =  videoUrl;
    }
