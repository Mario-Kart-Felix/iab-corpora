(function() {
	//config
	var redirectTime = 15; //seconds
	var countDownInterval;

	//private vars
	var timeLeft = redirectTime;
	var skipped = false;

	//set it on oas
	BI.OAS.interstitial_timeout = redirectTime*1000;
	//an array to contain functions that may need to
	//be called after the interstitial has been hidden
	BI.OAS.interstitial_queue = [];

	//called on docready
	var init = function() {
		$('.standard_leaderboard').hide();
		$('.info a').attr('href',window.location)
		$('#interstitial-countdown span').html(timeLeft);
		$('.ad-wallpaper').hide();
		$('#placed_wallpaper').hide();
		$('#main-content').hide();
		$('#eyeDiv').hide();

		countDownInterval = setInterval(countdown,1000);
		setTimeout(redirect,redirectTime*1000);

		//next tick we want to notify that the ad will be shown
		//(race-condition with other stuff firing on doc.ready and we want to make sure everything bound)
		setTimeout(function() {amplify.publish('interstitialAdShown');},0);
	}

	var countdown = function() {
		timeLeft--;
		$('#interstitial-countdown span').html(timeLeft < 0 ? 0 : timeLeft);

		if (skipped) { return false; }
		//remove interstitial, show content
		$('.interstitial-ad-content').remove();
		$('#main-content, #ft').show()

		$('#main-content').show();

		certain pushdown ads don't properly position the pushdown element if they
		are hidden when the ad was rendered.  We can force them to reposition
		themselves by firing a window resize event

		we do this on next event-loop (using setTimeout 0) so we know all the
		post-interstitial logic has already fired

		This is based off http://stackoverflow.com/questions/3856098/trigger-onresize-in-cross-browser-compatible-manner
		*/
		setTimeout(function() {
			if (document.createEvent) {
			    var e = document.createEvent('HTMLEvents');
			    e.initEvent('resize', true, false);
			    document.body.dispatchEvent(e);

			} else if (document.createEventObject) {
			    document.body.fireEvent('onresize');
			}
		},0)

		skipped = true;
		BI.OAS.interstitial_queue.forEach(function(cb) {
