	if ( document.getElementById ) {
		var nodePtrs = [];
		var randomHtml = [];
		
		// copy nodes innerHTML
		for ( var i = 0, limit = nodeIds.length; i < limit; i++ ) {
			if ( document.getElementById( nodeIds[i] ) && document.getElementById( nodeIds[i] ).innerHTML ) {
				nodePtrs[i] = document.getElementById( nodeIds[i] );
				randomHtml[i] = nodePtrs[i].innerHTML;
			}
		}
		
		// randomize
		randomHtml.sort( function ( a, b ) { return ( Math.random() < 0.5 ? -1 : 1 ); } );
		
		// write innerHTML back in

			nodePtrs[i].innerHTML = randomHtml[i];
		}
	}
