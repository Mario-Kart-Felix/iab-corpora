{
    this.video = video;
    this.playerId = playerId;
    this.width = width;
    this.height = height;
    this.autoPlay = autoPlay;

        if (!this.runNow()) {
            var self = this;

        }

        if (typeof xml_url != 'undefined') {
            if (this.timer)
                clearInterval(this.timer);

        }
        return false;
    }
}
