buildXHTMLElement = function(data) {
var el = [];
var open = function (data) {
var tag = data.tag;
var attrs = (!data.attrs) ? false : data.attrs;
el.push('<');
el.push(tag);
if (attrs) {
for (var attr in attrs) {
el.push(' ', attr.replace('Name', ''), '=', '"', attrs[attr], '"');
}
}
el.push('>');
return el;
};
var addContent = function(data) {
var content = data.content; // content of DOM element
el.push(content);
return el;
};
var close = function(data) {
var tag = data.tag;
el.push('</', tag, '>');
return el;
};
open(data);
addContent(data);
close(data);
return el.join('');
},
