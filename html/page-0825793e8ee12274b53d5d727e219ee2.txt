	function loadmodels(){
		f = document.getElementById("selectMake").value;
		ff = $('#selectMake').val();
		g = document.getElementById("selectModel");
		
		clearOptions(g);
		
		for( w=0; w<ff.length; w++ ){
			for( i=0; i<models[(ff[w])].length; i++ ){
				h = models[(ff[w])][i];
				c = models_count[(ff[w])][i];
				AddItem(h,h,c);
			}
		}

	}

	function AddItem(Text,Value,Count){
		// create an option object
		var opt = document.createElement("option");
		
		document.getElementById("selectModel").options.add(opt);
		opt.text = Text + " (" + Count + ")";
		opt.value = Value;
	}
	
	function clearOptions(oOptions){
		for( var i=oOptions.options.length-1;i>=0;i--){
			oOptions.options[i] = null;
		}
		oOptions.selectIndex = -1;
		AddItem("View All","View All","%");
	}
	
	function search_vehicles(){
		// add all search values to search form
		var eachMake = [];
		var eachModel = [];
		
		$('#selectMake :selected').each(function(i, selected){ eachMake[i] = $(selected).text(); });
		$('#selectModel :selected').each(function(i, selected){ eachModel[i] = $(selected).text(); });
		
		var makeString = "'";
		for( i=0; i<eachMake.length; i++ ){
			if( i == (eachMake.length-1) ){
				makeString += eachMake[i];
			} else {
				makeString += eachMake[i] + ",";
			}
		}
		makeString += "";
		
		var modelString = "";
		
		for( i=0; i<eachModel.length; i++ ){
			if( i < eachModel.length ){
				modelString += eachModel[i] + ",";
			}
			else{
				modelString += eachModel[i];
			}
		}
		modelString += "";
		
		// passing these values incased in single quotes to make sure it posts right
		$('#s_make').attr("value",makeString);
		$('#s_model').attr("value",modelString);
		$('#s_min_y').attr("value",$('#minYear').attr("value"));
		$('#s_max_y').attr("value",$('#maxYear').attr("value"));
		$('#s_min_p').attr("value",$('#minPrice').attr("value"));
		$('#s_max_p').attr("value",$('#maxPrice').attr("value"));
		
		// alert($('#s_make').val() + " " + $('#s_model').val() );

		$('#PostSearchForm').submit();
		// and submit it
		
	}
	
	function newused(which){

		h = document.getElementById("selectMake");

			case "new":
				$('#used_make').hide();

				$('#used_make').show();
				g = document.getElementById("selectModel");

				$('#new_make').show();

				$('#NewUsedField').attr("value","New");
				clearOptions(g);
		}
		return;
