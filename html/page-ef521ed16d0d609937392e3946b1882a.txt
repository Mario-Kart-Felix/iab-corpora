        PAGESETUP.files.push('http://static.ed.edmunds-media.com/edmunds-webassets/1.47.60/wro/edmui.droplist-configurator-proto.js');
        PAGESETUP.addControl(function() {
            var $D = YAHOO.util.Dom, $E = YAHOO.util.Event;

            function copyOptions(from, to){
                var targMake = $D.get(to);
                targMake.length = 0;
                var srcMake = $D.get(from);
                for (var i=0; i<srcMake.length; i++) {
                    targMake.options[i] = new Option(srcMake.options[i].text,srcMake.options[i].value);
                }
            }
            copyOptions('new_makes_options','uci_make_dd');
            document.uci_mm_select_form.makes.selectedIndex=0;
            document.uci_mm_select_form.models.length=1;
            $D.get('uci_ptype_new').checked=false;
            $D.get('uci_ptype_used').checked=true;
            $D.addClass('uci_hp_widget', 'tid');

            var zipCode = $D.get('zip_input').value;
            var eddirectOnly = "true";
            var droplist = new EDMUNDS.DroplistConfigurator();
            $E.addListener('uci_ptype_new','click',function(){
                copyOptions('new_makes_options','uci_make_dd');
                $D.removeClass('uci_make_dd', 'nodisplay');
                $D.removeClass('uci_model_dd', 'nodisplay');
                $D.addClass('uci_used_make_dd', 'nodisplay');
                $D.addClass('uci_used_model_dd', 'nodisplay');
                $D.removeClass('uci_hp_widget', 'tid');
                document.uci_mm_select_form.makes.selectedIndex=0;
                document.uci_mm_select_form.models.length=1;
                document.uci_mm_select_form.used_makes.selectedIndex=0;
                document.uci_mm_select_form.submodels.length=1;
            });
            $E.addListener('uci_ptype_used','click',function(){
                $D.addClass('uci_make_dd', 'nodisplay');
                $D.addClass('uci_model_dd', 'nodisplay');
                $D.removeClass('uci_used_make_dd', 'nodisplay');
                $D.removeClass('uci_used_model_dd', 'nodisplay');
                $D.addClass('uci_hp_widget', 'tid');
                document.uci_mm_select_form.makes.selectedIndex=0;
                document.uci_mm_select_form.models.length=1;
                document.uci_mm_select_form.used_makes.selectedIndex=0;
                document.uci_mm_select_form.submodels.length=1;
                document.uci_mm_select_form.submodels.setAttribute("disabled", "disabled")
            });

            droplist.container = document.uci_mm_select_form;
            droplist.jsonResults = {};
            droplist.wsUrl = '/api/vehicle-directory-ajax/findmakes?fmt=json';

            droplist.droplistPopulator = function(jsonResults, opName, opValue, selectId, initOptions, selectedItem){
                var json = jsonResults;
                selectId.innerHTML ='';
                selectId.options[0] = new Option(initOptions.text, '');

                //selectId.innerHTML ='';
                var k = 1, hasUsed, hasNew;

                for (property in json) {//if object
                    if (json.hasOwnProperty(property)) {

                        if (selectId.name=="models") {

                            hasNew = json[property]['years']['NEW'];
                            hasUsed = json[property]['years']['USED'];

                            if (($D.get('uci_ptype_new').checked && hasNew) || ($D.get('uci_ptype_used').checked && hasUsed)) {
                                selectId.options[k] = new Option(json[property][opName],json[property][opValue]);
                                selectId.options[k].setAttribute("model", json[property]["modelname"]);
                                if (json[property]["submodel"]) {
                                    selectId.options[k].setAttribute("sub", json[property]["submodel"]);
                                }
                                k++;
                            }
                        } else {
                            selectId.options[k] = new Option(json[property][opName],json[property][opValue]);
                            k++;
                        }

                    }
                }
                if(selectId.options.length > 1 && json.length){//if array
                    var count = json.length;
                    selectId.options[0] = new Option(initOptions.text, '');
                    var k = 1;
                    for(var i=1; i < count; i++){
                        selectId.options[k] = new Option(json[i][opName],json[i][opValue]);
                        k++;
                    }
                }

                selectId.removeAttribute("disabled");
                return selectId.innerHTML;//return for caching
            };
            droplist.init= function(){

                //store the form elements as vars
                this.make_droplist = this.container.makes;
                this.model_droplist = this.container.models;
                this.used_make_droplist = this.container.used_makes;
                this.used_model_droplist = this.container.submodels;
                this.submit_button = YAHOO.util.Selector.query('button.uci_droplist_submit', this.container, true);

                //disable droplists
                this.model_droplist.setAttribute("disabled","disabled");
                this.used_model_droplist.setAttribute("disabled","disabled");

                //store initial value in cache
                this.cache[this.make_droplist.name] = this.make_droplist[0];
                this.cache[this.model_droplist.name] = this.model_droplist[0];
                this.cache[this.used_make_droplist.name] = this.used_make_droplist[0];
                this.cache[this.used_model_droplist.name] = this.used_model_droplist[0];

                //attach listeners
                YAHOO.util.Event.addListener(this.container, "change", this.onChange, this);
                YAHOO.util.Event.addListener(this.container, "submit", this.onSubmit, this);
                YAHOO.util.Event.addListener(this.submit_button, "click", this.onSubmit, this);

                //IE specific onchange event doesnt fire unless specifically referencing the node
                if(YAHOO.env.ua.ie > 0){
                    YAHOO.util.Event.addListener(this.make_droplist, "change", this.onChange, this);
                    YAHOO.util.Event.addListener(this.model_droplist, "change", this.onChange, this);
                    YAHOO.util.Event.addListener(this.used_make_droplist, "change", this.onChange, this);
                    YAHOO.util.Event.addListener(this.used_model_droplist, "change", this.onChange, this);
                }

                //create initial droplist specific to makes json result only if droplist not populated
                if(!this.make_droplist.options[1]){
                    if(this.wsUrl){
                        obj = this;
                        YAHOO.util.Connect.asyncRequest('GET', this.wsUrl, {
                            success:function(o) {
                                if(o.responseText !== undefined){
                                    var jsonList = YAHOO.lang.JSON.parse(o.responseText);
                                    obj.droplistPopulator(jsonList.makes, 'name', 'name', obj.make_droplist, {text:'Select make'});
                                }
                            },
                            failure:function(o) {	}
                        });
                    }else{
                        var jsonList = this.jsonResults.makes;
                        this.droplistPopulator(jsonList, 'name', 'name', this.make_droplist, {text:'Select make'});
                    }
                }
                //init used makes
                var usedMakesUrl = "/api/inventory/models/?zipcode=" + zipCode + "&radius=100";
                if (eddirectOnly == "true") {
                    usedMakesUrl = usedMakesUrl + "&eddirect_only=true";
                }
                var object = this;
                YAHOO.util.Connect.asyncRequest('GET', usedMakesUrl, {
                    success:function(o) {
                        if(o.responseText != undefined) {
                            var jsonList = YAHOO.lang.JSON.parse(o.responseText);
                            if (jsonList.hasOwnProperty("makes")) {
                                var usedMakesList = jsonList["makes"];
                                for (k = 0; k < usedMakesList.length; k++) {
                                    object.used_make_droplist.options[k+1] = new Option(usedMakesList[k], usedMakesList[k]);
                                }
                            }
                        }
                    },
                    failure: function(o) {  }
                })

            };
            droplist.onChange= function(e,obj){
                var target = YAHOO.util.Event.getTarget(e);
                var isNew = $D.get('uci_ptype_new').checked;
                switch(target.name){
                    case "makes":
                        if(target.value){
                            var makeId = obj.make_droplist.value;
                            var ps = '&ps=new';
                            var sUrl = "/api/vehicle-directory-ajax/findmakemodels?make="+obj.make_droplist.value+"&fmt=json&yearFormat=expanded"+ps;

                            obj.model_droplist.options[0] = new Option(obj.cache[obj.model_droplist.name].text,'');

                            //get json and update droplist
                            YAHOO.util.Connect.asyncRequest('GET', sUrl, {
                                success:function(o) {
                                    if(o.responseText !== undefined){
                                        //not the prettiest
                                        jsonList = YAHOO.lang.JSON.parse(o.responseText);
                                        for (var sub in jsonList["models"]) {
                                            if (jsonList["models"].hasOwnProperty(sub)) {
                                                jsonList["models"][sub].id = sub;
                                            }
                                        }
                                        obj.jsonResults = jsonList["models"]; //update jsonResults with model/year info
                                        obj.jsonModelYearPersist = obj.jsonModelYearPersist || {};
                                        obj.jsonModelYearPersist[makeId] = jsonList["models"];
                                        obj.cache[makeId] = obj.droplistPopulator(jsonList["models"], 'name', 'id', obj.model_droplist,  {text:'Select model'});
                                        return true;
                                    }
                                },
                                failure:function(o) {	}
                            });
                        }else{
                            //reset to initial state
                            obj.model_droplist.setAttribute("disabled","disabled");
                            obj.model_droplist.innerHTML='';
                            obj.model_droplist.options[0] = new Option(obj.cache[obj.model_droplist.name].text,'');
                        }
                        break;
                    case "used_makes":

                            var makeId = obj.used_make_droplist.value;

                                    if (jsonList.hasOwnProperty("submodels")) {
                                        var submodelList = jsonList["submodels"];
                                        for (k = 0; k < submodelList.length; k++) {
                                            obj.used_model_droplist.options[k+1] = new Option(submodelList[k], submodelList[k]);
                                        }
                                        obj.used_model_droplist.removeAttribute("disabled")
                                    }
                                },
                                failure:function(o) {
                                    obj.used_model_droplist.setAttribute("disabled","disabled");
                                }
                            });
                        } else {
                            obj.used_model_droplist.setAttribute("disabled","disabled");
                            obj.used_model_droplist.innerHTML='';
                            obj.used_model_droplist.options[0] = new Option(obj.cache[obj.used_model_droplist.name].text,'');
                        }
                        break;
                }
            };

                    for (i = 0; i <len; i++) {

            };
