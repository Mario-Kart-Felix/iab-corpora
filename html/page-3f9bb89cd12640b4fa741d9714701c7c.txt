function getAHUrl(autoHomeValue, autoHomeStateMenu) {

	var homeValue = "home";

function getAllIDs() {
	var cookiePName = 'ahPartnerID';
	var queryPString = 'PartnerID';
	var queryPStringValue = gup(queryPString);
	var explicitPartnerID = "";

	var cookieRName = 'ahRefID';
	var queryRString = 'RefID';
	var queryRStringValue = gup(queryRString);

	var cookieEName = 'ahEmailID';
	var queryEString = 'EmailID';
	var queryEStringValue = gup(queryEString);
	
	var allIDs = "";

	if (explicitPartnerID != "") {
			$.cookie(cookiePName,explicitPartnerID,{ path: '/', domain: 'ameriprise.com'  });
	}

	if (queryPStringValue != "" || queryRStringValue != "" || queryEStringValue != "") {
		
		if (queryPStringValue != "") { 
			$.cookie(cookiePName,queryPStringValue,{ path: '/', domain: 'ameriprise.com' });
		}
		
		if (queryRStringValue != "") {
			$.cookie(cookieRName,queryRStringValue,{ path: '/', domain: 'ameriprise.com' });
		} else {
			$.cookie(cookieRName,'',{ path: '/', domain: 'ameriprise.com' });
		}
		
		if (queryEStringValue != "") {
			$.cookie(cookieEName,queryEStringValue,{ path: '/', domain: 'ameriprise.com' });
		} else {
			$.cookie(cookieEName,'',{ path: '/', domain: 'ameriprise.com' });
		}
				
	}
	
	allIDs = getPartnerRefEmailIDs();

	return allIDs;

}

function getPartnerID() { 
	var cookieName = "ahPartnerID"; 
	var ahPartnerID = ""; 
	var tempURL = window.location.href.toLowerCase();
	if ($.cookie(cookieName)) {
		ahPartnerID = $.cookie(cookieName);
	    } else if (tempURL.indexOf("costco") > -1) {
	                ahPartnerID = "COSTCO";
	    } else if (tempURL.indexOf("ford") > -1) {
	                ahPartnerID = "Ford";
	    }
		return ahPartnerID; 
}

function isPartnerEqualTo(partnerName) {
	var isPartnerEqualTo = false;
	
	if (getPartnerID() === partnerName || gup('PartnerID') === partnerName) {
		isPartnerEqualTo = true;
	}
	
	return isPartnerEqualTo;
}


function isValueInArray(value, array) {
	var isValueInArray = false;
	
	for (var i=0 ; i < array.length; i++) {
		if (value === array[i]) {
			isValueInArray = true;
		}
	}
	
	return isValueInArray;
}

function getRadioButtonValue(radioButtonList) {
	var radioButtonValue = "";
	
	for (var i = 0; i < radioButtonList.length; i++) {
		if (radioButtonList[i].checked) {
			radioButtonValue = radioButtonList[i].value;
		}
	}
	
	return radioButtonValue;
}

function gup( name )
{
  name = name.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");
  var regexS = "[\\?&]"+name+"=([^&#]*)";
  var regex = new RegExp( regexS );
  var results = regex.exec( window.location.href );
  if( results == null )
    return "";
  else
    return results[1];
}

	$('a').each(function(){
    var thisHref = $(this).attr('href');
		var hrefConnector = "&" + allIDs;
		if (thisHref.indexOf('()') != -1 || thisHref.indexOf('mailto') != -1 || thisHref.indexOf('#') != -1) {
			hrefConnector = "";
		} else if (thisHref.indexOf('?') == -1) {
			hrefConnector = "?" + allIDs;
		}
		
    var newHref = thisHref + hrefConnector;
    
    $(this).attr('href', newHref);
	});
	
	if (isPartnerEqualTo("COSTCO")) {
		$("html").addClass("costco");
	}
	if (isPartnerEqualTo("Ford")) {
		$("html").addClass("ford");
	}
